58      public MIDITrack parseChord(String s) {
59          MIDITrack list = new MIDITrack();
60          Scanner sc = new Scanner(s);
61          sc.useDelimiter("+");
62          while (sc.hasNext()) {
63              String p = sc.next();
64              MIDINote n = parseNote(p);
65              list.add(n);
66          }
67          sc.close();
68          return list;
69      }

71      public static String createString(MIDITrack notelist) {
72          // Pitch, Start beat, Duration, Velocity, Pan, Channel, Bank, Pitchbend
73          StringBuilder sb = new StringBuilder();
74          for (MIDINote n : notelist) {
75              sb.append(n.getPitch().toString().trim()).append(',');
76              sb.append(n.getStartBeat()).append(',');
77              sb.append(n.getDuration()).append(',');
78              sb.append(n.getVelocity()).append(',');
79              sb.append(n.getPan()).append(',');
80              sb.append(n.getChannel()).append(',');
81              sb.append(n.getBank()).append(',');
82              sb.append(n.getProgram()).append(',');
83              sb.append(n.getPitchBend()).append(' ');
84          }
85          return sb.toString();
86      }

156 	 * Pitch, Start beat, Duration, Velocity, Pan, Channel, Bank, Program

163     public MIDINote parseNote(String s) {
164         MIDINote n = null;
165         Scanner sc = new Scanner(s);
166         sc.useDelimiter(",");
167 
168         String p = null;
169 
170         if (sc.hasNext(fullPitchPattern)) {
171             p = sc.next();
172         } else if (sc.hasNext(shortPitchPattern)) {
173             p = sc.next();
174             p += runningOctave;
175             logger.debug("short " + p);
176         } else {
177             String tok = sc.next();
178             sc.close();
179             throw new MIDIParserException(String
180                     .format("Bad pitch token '%s'",
181                             tok));
182         }
183 
184         double start = 1d;
185         double duration = 1d;
186 
187         if (sc.hasNextDouble()) {
188             start = sc.nextDouble();
189         }
190 
191         // couldBeDurationString(tok);
192 
193         if (sc.hasNext(DurationParser.allPattern)) {
194             String tok = sc.next(DurationParser.allPattern);
195             logger.debug("token {}", tok);
196             duration = DurationParser.getDuration(tok);
197         } else if (sc.hasNextDouble()) {
198             duration = sc.nextDouble();
199         }
200 
201         n = new MIDINote(p, start, duration);
202         runningOctave = n.getPitch().getMidiNumber() / 12;
203         logger.debug(
204                 String.format("p %s note %s ro %d",
205                         p,
206                         n,
207                         runningOctave));
208 
209         logger.debug("pitch {} note {} running octave {}",
210                 p,
211                 n,
212                 runningOctave);
213 
214         if (sc.hasNextInt()) {
215             int velocity = sc.nextInt();
216             n.setVelocity(velocity);
217         }
218 
219         if (sc.hasNextInt()) {
220             int pan = sc.nextInt();
221             n.setPan(pan);
222         }
223 
224         if (sc.hasNextInt()) {
225             int channel = sc.nextInt();
226             n.setChannel(channel);
227         }
228 
229         if (sc.hasNextInt()) {
230             int bank = sc.nextInt();
231             n.setBank(bank);
232         }
233         if (sc.hasNextInt()) {
234             int program = sc.nextInt();
235             n.setProgram(program);
236         }
237 
238         if (sc.hasNextInt()) {
239             int pb = sc.nextInt();
240             n.setPitchBend(pb);
241         }
242         sc.close();
243         return n;
244     }

320     public String checkMulti(String s) {
321         Scanner sc = new Scanner(s);
322         String patternString = " *X(\\d)\\s+\\((.+?)\\)+?";
323         Pattern multiPattern = Pattern
324                 .compile(patternString);
325         int numRepeats = 0;
326         StringBuilder origsb = new StringBuilder(s);
327 
328         logger.debug("checking for multi '{}'", s);
329         sc.useDelimiter("\\Z");
330         // logger.debug("multi next '{}'", sc.next());
331         String multi = sc.findInLine(multiPattern);
332         logger.debug("multi '{}'", multi);
333         if (multi != null) {
334             numRepeats = Integer.parseInt(sc.match().group(1));
335             int start = sc.match().start();
336             int end = sc.match().end();
337 
338             logger.debug("multi group 1 '{}'", sc.match().group(1));
339             logger.debug("multi group 2 '{}'", sc.match().group(2));
340             logger.debug("multi group count '{}'", sc.match()
341                     .groupCount());
342             logger.debug("multi start {} end {}", start, end);
343 
344             StringBuilder sb2 = new StringBuilder();
345             sb2.append(' ');
346             for (int i = 0; i < numRepeats; i++) {
347                 sb2.append(sc.match().group(2)).append(' ');
348             }
349 
350             logger.debug("origsb before '{}'", origsb);
351             origsb.replace(start, end, sb2.toString());
352             logger.debug("origsb after '{}'", origsb);
353 
354             Matcher matcher = multiPattern.matcher(origsb);
355             boolean recurse = matcher.find();
356             logger.debug("recurse {}", recurse);
357             if (recurse) {
358                 origsb = new StringBuilder(checkMulti(origsb.toString()));
359             }
360         } else {
361             logger.debug("no multi in '{}'", s);
362         }
363         sc.close();
364         return origsb.toString().trim();
365     }
