101     static void stdout(Document doc) {
102         logger.debug("Enter stdout");

105             StreamResult streamResult = new StreamResult(System.out);

114     public static void writeScaleXML(OutputStream out) {
115         logger.debug("enter writeScaleXML");
116         DocumentBuilderFactory dbf = null;
117         DocumentBuilder db = null;
118         try {
119             dbf = DocumentBuilderFactory.newInstance();
120             dbf.setValidating(false);
121             db = dbf.newDocumentBuilder();
122         } catch (ParserConfigurationException e) {
123             e.printStackTrace();
124             logger.error(e.getLocalizedMessage(), e);
125             return;
126         }
127         Document doc = db.newDocument();
128 
129         Element chords = doc.createElement("scales");
130         doc.appendChild(chords);
131 
132         // Element extends Node
133 
134         List<String> list = ScaleFactory.getScaleNames();
135 
136         for (String name : list) {
137             Scale scale = ScaleFactory.getScaleByName(name);
138             logger.debug("writing scale " + scale);
139             Element scaleElement = doc.createElement("scale");
140             chords.appendChild(scaleElement);
141             scaleElement.setAttribute("name", scale.getName());
142 
143             Element intervals = doc.createElement("intervals");
144             scaleElement.appendChild(intervals);
145             for (Integer i : scale.getIntervals()) {
146                 Element intervalel = doc.createElement("interval");
147                 String interval = i.toString();
148                 intervalel.appendChild(doc.createTextNode(interval));
149                 intervals.appendChild(intervalel);
150             }
151 
152             if (scale.isDescendingDifferent()) {
153                 logger.debug("scale descending is different");
154                 Element dintervals = doc.createElement("descendingIntervals");
155                 scaleElement.appendChild(dintervals);
156                 for (Integer i : scale.getDescendingIntervals()) {
157                     Element intervalel = doc.createElement("interval");
158                     String interval = i.toString();
159                     intervalel.appendChild(doc.createTextNode(interval));
160                     dintervals.appendChild(intervalel);
161                 }
162             }
163 
164             Element spelling = doc.createElement("spelling");
165             scaleElement.appendChild(spelling);
166             String s = scale.getSpelling();
167             String[] spellings = s.split("\\s");
168             for (String degree : spellings) {
169                 Element degreeEl = doc.createElement("degree");
170                 degreeEl.appendChild(doc.createTextNode(degree));
171                 spelling.appendChild(degreeEl);
172             }
173 
174             String descr = scale.getDescription();
175             if (descr != null) {
176                 Element el = doc.createElement("description");
177                 scaleElement.appendChild(el);
178                 el.appendChild(doc.createTextNode(descr));
179             }
180         }
181         // logger.debug("writing to stdout");
182         // try {
183         // DOMSource domSource = new DOMSource(doc);
184         // StreamResult streamResult = new StreamResult(out);
185         // TransformerFactory tf = TransformerFactory.newInstance();

217     public static void readScales(String filename) {
218         // these are jaxp classes
219         DocumentBuilder db = null;
220         try {
221             DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
222 
223             // dbf.setCoalescing(true);
224             // dbf.setExpandEntityReferences(true);
225             // dbf.setIgnoringComments(false);
226             // dbf.setIgnoringElementContentWhitespace(true);
227             // dbf.setNamespaceAware(true);
228 
229             dbf.setValidating(false);
230             db = dbf.newDocumentBuilder();
231             // db.setEntityResolver(EntityResolver er);
232             // db.setErrorHandler(ErrorHandler eh);
233 
234         } catch (ParserConfigurationException e) {
235             logger.error(e.getLocalizedMessage(), e);
236             e.printStackTrace();
237             return;
238         }
239 
240         Document doc = null;
241 
242         try {
243 
244             // all of this nonsense in case you're using this from
245             // a swing app or from a servlet. A servlet would set the
246             // inputstream
247             // in its init method.
248             InputSource is = null;
249 
250             if (inputStream != null) {
251                 logger.debug("reading from inputstream");
252                 is = new InputSource(inputStream);
253             } else {
254 
255                 File f = new File(filename);
256                 // String uri = f.toURL().toString();
257                 String uri = f.toURI().toURL().toString();
258 
259                 logger.debug("defs ", scaleDefinitionFileName);
260                 logger.debug("uri ", uri);
261 
262                 // filereader always uses default encoding
263                 // is = new InputSource(new FileReader(f));
264                 is = new InputSource(new InputStreamReader(new FileInputStream(
265                         f), Charset.forName("ISO-8859-1")));
266 
267                 // is = new InputSource(ScaleFactoryXMLHelper.class
268                 // .getResourceAsStream(uri));
269                 // if not set, then it won't show up in exceptions
270                 is.setSystemId(uri);
271             }
272 
273             doc = db.parse(is);
274             doc.normalize();
275         } catch (SAXParseException spe) {
276             logger.error(spe.getLocalizedMessage(), spe);
277             Exception x = spe.getException();
278             if (x != null) {
279                 logger.error("error parsing", x);
280             }
281             return;            
282         } catch (SAXException sxe) {
283             Exception x = sxe;
284             if (sxe.getException() != null) {
285                 x = sxe.getException();
286                 logger.error(x.getLocalizedMessage(), x);
287             }
288             return;
289             // } catch (MalformedURLException mue) {
290 
291         } catch (IOException ioe) {
292             logger.error(ioe.getLocalizedMessage(), ioe);
293             return;
294 
295         } catch (Throwable t) {
296             t.printStackTrace();
297             logger.error(t.getLocalizedMessage(), t);
298             return;
299         }
300 
301         Element e = doc.getDocumentElement();
302         String name = e.getTagName();
303         logger.debug("parsing scales document element {}", name);
304 
305         NodeList scales = e.getElementsByTagName("scale");
306         if (scales != null) {
307             int length = scales.getLength();
308             for (int i = 0; i < length; i++) {
309                 Node nl = scales.item(i);
310                 logger.debug("parsing scale ");
311                 parseScale(nl);
312             }
313         } else {
314             logger.error("no scales!");            
315         }
316     }

318     /**
319      * Attempt to parse each scale element and register it with the
320      * ScaleFactory. Ignores the spelling if the intervals are specified.
321      * 
322      * ScaleFactory.registerScale(mc);
323      * 
324      * @param scale
325      */
326     private static void parseScale(Node scale) {
327         if (scale == null) {
328             logger.error("parseScale: node is null");
329             return;
330         }
331         if (scale.getNodeName().equals("scale") == false) {
332             logger.error("parseScale: node is not a scale");
333             return;
334         }
335         NamedNodeMap pa = scale.getAttributes();
336         Node ni = pa.getNamedItem("name");
337         if (ni == null) {
338             throw new IllegalArgumentException(
339                     "Invalid scale; no name attribute");
340         }
341         String name = ni.getNodeValue();
342         logger.debug(name);
343         Integer[] intervals = null;
344         Integer[] dintervals = null;
345         String spelling = null;
346         String description = null;
347 
348         if (scale.hasChildNodes()) {
349             NodeList sc = scale.getChildNodes();
350             int len = sc.getLength();
351             for (int i = 0; i < len; i++) {
352                 Node nn = sc.item(i);
353                 if (nn.getNodeName().equalsIgnoreCase("intervals")) {
354                     intervals = parseIntervals(nn);
355                 }
356                 if (nn.getNodeName().equalsIgnoreCase("descendingIntervals")) {
357                     dintervals = parseDescendingIntervals(nn);
358                 }
359                 if (nn.getNodeName().equalsIgnoreCase("spelling")) {
360                     spelling = parseSpelling(nn);
361                 }
362                 if (nn.getNodeName().equalsIgnoreCase("description")) {
363                     description = nn.getTextContent();
364                     logger.debug("description " + description);
365                 }
366             }
367         }
368         // String s = String.format("xml %s has intervals %s \n", name,
369         // ArrayUtils
370         // .asString(intervals));
371         // System.err.println(s);
372 
373         Scale mc = null;
374         if (intervals != null) {
375             mc = new Scale(name, intervals, description);
376             logger.debug("creating from intervals " + name);
377             if (spelling != null) {
378                 mc.setSpelling(spelling);
379             }
380         } else if (spelling != null) {
381             mc = new Scale(name, spelling);
382             mc.setDescription(description);
383         }
384 
385         if (dintervals != null) {
386             mc.setDescendingIntervals(dintervals);
387         }
388         if (mc != null) {
389             ScaleFactory.registerScale(mc);
390         } else {
391             logger.error("Scale was null. Not registered");
392         }
393     }

395     private static Integer[] parseIntervals(Node intervals) {
396         NodeList sc = intervals.getChildNodes();
397         int len = sc.getLength();
398         // int[] a = new int[len];
399         List<Integer> list = new ArrayList<Integer>();
400 
401         for (int i = 0; i < len; i++) {
402             Node nn = sc.item(i);
403             if (nn.getNodeName().equalsIgnoreCase("interval")) {
404                 // System.out.println(nn.getTextContent());
405                 // a[i] = Integer.parseInt(nn.getTextContent());
406                 list.add(Integer.parseInt(nn.getTextContent()));
407             }
408         }
409         Integer[] array = new Integer[list.size()];
410         return list.toArray(array);
411     }

421     private static Integer[] parseDescendingIntervals(Node intervals) {

425         List<Integer> list = new ArrayList<Integer>();

433             }
434         }
435         Integer[] array = new Integer[list.size()];
436         return list.toArray(array);

439     private static String parseSpelling(Node spelling) {
440         StringBuilder sb = new StringBuilder();
441         NodeList sc = spelling.getChildNodes();
442         int len = sc.getLength();
443         for (int i = 0; i < len; i++) {
444             Node nn = sc.item(i);
445             if (nn.getNodeName().equalsIgnoreCase("degree")) {
446                 sb.append(nn.getTextContent());
447                 sb.append(' ');
448             }
449         }
450         return sb.toString();
451     }
