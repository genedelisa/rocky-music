392     /**
393      * @param track
394      *            append to this JavaSound Track
395      * @param chan
396      *            on this channel
397      * @param d1
398      *            data1
399      * @param d2
400      *            data2
401      */
402     public static void appendControlChange(final Track track, final int chan,
403             final int d1, final int d2) {
404         try {
405             ShortMessage message = new ShortMessage();
406             long tick = track.ticks();
407             // message.setMessage(0xB0 + chan, d1, d2);
408             message.setMessage(ShortMessage.CONTROL_CHANGE,
409                     chan,
410                     d1,
411                     d2);
412             MidiEvent event = new MidiEvent(message, tick);
413             track.add(event);
414         } catch (Exception ex) {
415             ex.printStackTrace();
416         }
417     }

419     /**
420      * @param value
421      *            the controller number
422      * @return the controller name
423      */
424     public static String getControllerName(final Integer value) {
425         String r = "Unknown controller: " + value;
426         for (Entry<String, Integer> e : CONTROLLER_NAMES.entrySet()) {
427             if (e.getValue().equals(value)) {
428                 r = e.getKey();
429                 break;
430             }
431         }
432         return r;
433     }
434 
435     /**
436      * @param name
437      *            controller name
438      * @return the controller number
439      */
440     public static int getControllerNumber(final String name) {
441         Integer i = CONTROLLER_NAMES.get(name);
442         if (i == null) {
443             System.err.println("name is null");
444             throw new IllegalArgumentException(name);
445         }
446         return i.intValue();
447     }

449     /**
450      * @param track

452      * @param tick
453      *            when to insert it
454      * @param chan
455      *            on which channel

457      *            data1
458      * @param d2
459      *            data2
460      */
461     public static void insertControlChange(final Track track, final long tick,
462             final int chan,
463             final int d1, final int d2) {
464         try {
465             ShortMessage message = new ShortMessage();
466             // message.setMessage(0xB0 + chan, d1, d2);
467             message.setMessage(ShortMessage.CONTROL_CHANGE,
468                     chan,
469                     d1,
470                     d2);
471             MidiEvent event = new MidiEvent(message, tick);
472             track.add(event);
473         } catch (Exception ex) {
474             ex.printStackTrace();
475         }
476     }
