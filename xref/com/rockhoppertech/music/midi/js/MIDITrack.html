434     @Override
435     public String toString() {
436         StringBuilder sb = new StringBuilder();
437         sb.append("Track Name:").append(name).append('\n');
438         // sb.append("Instrument:").append(this.gmpatch).append('\n');
439         sb.append("Instrument:").append(this.instrument).append('\n');
440 
441         if (this.notes.size() > 0) {
442             for (MIDINote n : notes) {
443                 sb.append(n).append('\n');
444             }
445         } else {
446             sb.append("no notes").append('\n');
447         }
448 
449         if (this.events.size() > 0) {
450             sb.append("events").append('\n');
451             for (MIDIEvent n : events) {
452                 sb.append(n.toReadableString()).append('\n');
453                 sb.append(n.toString()).append('\n');
454             }
455         } else {
456             sb.append("no events").append('\n');
457         }
458 
459         NavigableMap<Double, KeySignature> keys = this.getKeySignatures();
460         for (Double time : keys.keySet()) {
461             KeySignature key = keys.get(time);
462             sb.append("Key: ").append(key.getDisplayName()).append(" at beat ")
463                     .append(time).append('\n');
464         }
465 
466         NavigableMap<Double, TimeSignature> timeSigs = this
467                 .getTimeSignatures();
468         for (Double time : timeSigs.keySet()) {
469             TimeSignature ts = timeSigs.get(time);
470             sb.append("Time signature: ").append(ts.getDisplayName())
471                     .append(" at beat ").append(time).append('\n');
472         }
473 
474         NavigableMap<Double, Integer> tempoMap = this.getTempoMap();
475         for (Double time : tempoMap.keySet()) {
476             Integer tempo = tempoMap.get(time);
477             sb.append("Tempo: ").append(tempo).append(" at beat ").append(time)
478                     .append('\n');
479         }
480 
481         return sb.toString();
482     }
