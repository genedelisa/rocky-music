28  /**
29   * A JavaSound receiver that prints to stdout.
30   * 
31   * @author <a href="mailto:gene@rockhoppertech.com">Gene De Lisa</a>
32   * 
33   */
34  public class ConsoleReceiver implements Receiver {
35      private boolean open = false;
36      // private int mode = OMNI_ON_POLY;
37      private int mode = 1;
38      private MidiDevice.Info info;
39  
40      double tempo = 120d;
41      double division = 480d;
42      double factor = (60000000d / tempo) / division;
43  
44      /**
45       * <p>
46       * We receive a message from the transmitter here. That's receive and not
47       * send. What a confusing name!!!!!! The timeStamp is in ms not ticks. Go
48       * figure. A -1 value means send immediately.
49       * </p>
50       * 
51       * @see javax.sound.midi.Receiver#send(javax.sound.midi.MidiMessage, long)
52       */
53      @Override
54      public void send(MidiMessage mm, long timeStamp) {
55  
56          System.out.println("ConsoleReceiver status 0x"
57                  + Integer.toHexString(mm.getStatus())
58                  + " timeStamp " + timeStamp);
59  
60          byte[] ba = mm.getMessage();
61          // skip active sensing
62          if ((ba[0] & 0xFF) == ShortMessage.ACTIVE_SENSING)
63              return;
64          StringBuilder sb = new StringBuilder();
65          sb.append("\ttimestamp: ");
66          sb.append(timeStamp);
67          if (timeStamp == -1) {
68              sb.append(" (Send immediately) ");
69          }
70          // print out the tick
71          long tick = (long) (timeStamp / factor);
72          sb.append(" tick:").append((timeStamp == -1 ? 0 : tick));
73          sb.append(" msg data:");
74          for (int i = 0; i < mm.getLength(); i++)
75              sb.append(" 0x").append(Integer.toHexString(ba[i] & 0xFF))
76                      .append(' ');
77  
78          System.out.println(sb.toString());
79  
80          if (mm instanceof ShortMessage) {
81              System.out.println("\t" + MIDIUtils.printFull((ShortMessage) mm));
82          }
83  
84          System.out.println();
85      }
86  
87      @Override
88      public void close() {
89          System.out.println("ConsoleReceiver closed");
90          this.open = false;
91      }
92  
93      public MidiDevice.Info getDeviceInfo() {
94          if (this.info == null) {
95              this.info = new Info("ConsoleReceiver", "Rockhopper Technologies",
96                      "Displays messages to stdout", "1.0");
97          }
98          return this.info;
99      }
