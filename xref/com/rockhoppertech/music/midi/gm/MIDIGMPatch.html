535     /**
536      * @param name the patch name
537      * @return a patch
538      */
539     public static MIDIGMPatch getPatch(String name) {
540         MIDIGMPatch match = null;
541         for (MIDIGMPatch patch : MIDIGMPatch.values()) {
542             if (patch.getName().equals(name)) {
543                 match = patch;
544                 break;
545             }
546         }
547         return match;
548     }

550     /**
551      * Match's the supplied program with the patche's program.
552      * @param program the program
553      * @return the name
554      */
555     public static String getName(int program) {
556         String match = null;
557         for (MIDIGMPatch patch : MIDIGMPatch.values()) {
558             if (patch.getProgram() == program) {
559                 match = patch.getName();
560                 break;
561             }
562         }
563         return match;
564     }
