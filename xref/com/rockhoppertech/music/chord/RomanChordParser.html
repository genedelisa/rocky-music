117     public static boolean isRoman(String input) {
118         // input = input.toUpperCase();
119         Matcher matcher = romanPattern.matcher(input);
120         logger.debug("checking {}", input);
121         matcher.reset();
122         boolean result = false;
123         if (matcher.find()) {
124             int length = matcher.end() - matcher.start();
125             if (length > 0)
126                 result = true;
127             String rs = input.substring(matcher.start(), matcher.end());
128             String val = String.format("start %d len %d: roman %s",
129                     matcher.start(), length, rs
130                     );
131             logger.debug("matched roman value: {}", val);
132 
133         } else {
134             logger.debug("no match for {}", input);
135         }
136         return result;
137     }

139     public static Pitch romanToPitch(Scale scale, String key, String romanString) {
140         romanString = romanString.trim();
141         romanString = romanString.toUpperCase(Locale.ENGLISH);
142         Matcher matcher = romanPattern.matcher(romanString);
143         int degree = 0;
144         int accidental = 0;
145         switch (romanString.charAt(0)) {
146         case '#':
147             accidental = 1;
148             break;
149         case 'b':
150             accidental = -1;
151             break;
152         case 'B':
153             accidental = -1;
154             break;
155         default:
156             accidental = 0;
157             break;
158         }
159         if (matcher.find()) {
160             // the rn without any chord symbol
161             String rs = romanString.substring(matcher.start(), matcher.end());
162             degree = romanToArabic(rs);
163             String val = String.format(
164                     "start %d len %d: roman '%s' = degree %d",
165                     matcher.start(),
166                     matcher.end() - matcher.start(),
167                     rs,
168                     degree);
169             if (logger.isDebugEnabled()) {
170                 String s = String.format("%s", val);
171                 logger.debug(s);
172             }
173         } else {
174             String s = String.format("Ain't no Roman numeral: '%s'",
175                     romanString);
176             throw new IllegalArgumentException(s);
177         }
178         List<Pitch> pits = scale.getDegreesAsPitches(key);
179         // account for zero based indexing
180         int idx = degree - 1;
181         if (idx < 0 || idx > pits.size()) {
182             throw new IllegalArgumentException(String.format(
183                     "I don't grok this: '%s'",
184                     romanString));
185         }
186         Pitch p = pits.get(idx);
187         if (logger.isDebugEnabled()) {
188             logger.debug(String.format("got pitch %s for degree %d, acc=%d", p,
189                     degree, accidental));
190         }
191         if (accidental != 0) {
192             p = p.transpose(accidental);
193         }
194         return p;
195     }

208      * @param key

211      *            a scale
212      * @param romanString
213      *            The roman numeral along with the music stuff
214      * @return a Chord
215      */
216     public static Chord getChord(String key, Scale scale, String romanString) {
217         String origRoman = romanString;
218         // romanString = romanString.toUpperCase().trim();
219         Pitch p = romanToPitch(scale, key, romanString);
220         String chordSymbol = null;
221         Matcher matcher = romanPattern.matcher(romanString);
222         if (matcher.find()) {
223             chordSymbol = origRoman.substring(matcher.end()).trim();
224             logger.debug(String.format("chordsymbol is '%s'", chordSymbol));
225         }
226         String name = PitchFormat.getPitchString(p);
227         if (chordSymbol != null && chordSymbol.equals("") == false) {
228             name += chordSymbol;
229         } else {
230             String[] symbols = ChordFactory.getChordSymbols(scale);
231             List<Pitch> pits = scale.getDegreesAsPitches(key);
232             int degree = pits.indexOf(p);
233             logger.debug(String.format("degree %d for %s", degree, p));
234             if (degree != -1 && symbols[degree] != null) {
235                 name += symbols[degree];
236             } else {
237                 if (logger.isDebugEnabled()) {
238                     logger.debug(String.format("no symbol for degree %d",
239                             degree));
240                 }
241             }
242         }
243         logger.debug(String.format("created name: %s", name));
244         Chord chord = null;
245         try {
246             chord = ChordFactory.getChordByFullSymbol(name);
247         } catch (Exception e) {
248             if (logger.isDebugEnabled()) {
249                 String s = String.format("bad '%s'", name);
250                 logger.error(s, e);
251             }
252         }
253         if (logger.isDebugEnabled()) {
254             String s = String.format("name '%s' returning chord: %s", name,
255                     chord.getDisplayName());
256             logger.debug(s);
257         }
258         return chord;
259     }

275     public static Chord getChord(int scaleRoot, String scaleName,
276             String romanString) {
277         Chord chord = null;
278         romanString = romanString.trim();
279         String origString = romanString;
280         romanString = romanString.toUpperCase(Locale.ENGLISH);
281         Matcher matcher = romanPattern.matcher(romanString);
282         int degree = 0;
283         String chordSymbol = null;
284         int accidental = 0;
285         switch (romanString.charAt(0)) {
286         case '#':
287             accidental = 1;
288             break;
289         case 'b':
290             accidental = -1;
291             break;
292         case 'B':
293             accidental = -1;
294             break;
295         default:
296             accidental = 0;
297             break;
298         }
299         if (matcher.find()) {
300             // the rn without any chord symbol
301             String rs = romanString.substring(matcher.start(), matcher.end());
302             logger.debug(String.format("roman numeral = '%s' is from '%s'", rs,
303                     romanString));
304             chordSymbol = origString.substring(matcher.end()).trim();
305             degree = romanToArabic(rs);
306             String val = String.format("start %d len %d: %s = %d", matcher
307                     .start(), matcher.end() - matcher.start(), rs, degree);
308             if (logger.isDebugEnabled()) {
309                 String s = String.format("%s", val);
310                 logger.debug(s);
311                 s = String.format("chord symbol: '%s'", chordSymbol);
312                 logger.debug(s);
313             }
314         } else {
315             String s = String.format("Ain't no Roman numeral: '%s'",
316                     romanString);
317             throw new IllegalArgumentException(s);
318         }
319 
320         List<Pitch> pits = ScaleFactory.getPitches(scaleName, scaleRoot);
321         // getDegreesAsPitches
322 
323         // account for zero based indexing
324         Pitch p = pits.get(degree - 1);
325         if (accidental != 0) {
326             p = p.transpose(accidental);
327         }
328         String name = PitchFormat.getPitchString(p);
329         if (chordSymbol.equals("") == false) {
330             name += chordSymbol;
331         } else {
332             Scale scale = ScaleFactory.createFromName(scaleName);
333             String[] symbols = ChordFactory.getChordSymbols(scale);
334             name += symbols[degree - 1];
335 
336             // if (scaleName.equalsIgnoreCase("Major")) {
337             // name += majorChords[degree - 1];
338             // } else {
339             // name += minorChords[degree - 1];
340             // }
341         }
342         try {
343             chord = ChordFactory.getChordByFullSymbol(name);
344         } catch (Exception e) {
345             if (logger.isDebugEnabled()) {
346                 String s = String.format("bad '%s'", name);
347                 logger.error(s, e);
348             }
349         }
350         if (logger.isDebugEnabled()) {
351             String s = String.format("name '%s' returning chord: %s", name,
352                     chord.getDisplayName());
353             logger.debug(s);
354         }
355         return chord;
356     }

385     static boolean containsPitchClass(List<Pitch> list, int pc) {
386         boolean contains = false;
387         for (Pitch p : list) {
388             if (p.getPitchClass() == pc) {
389                 contains = true;
390                 break;
391             }
392         }
393         return contains;
394     }

396     static int indexOfPitchClass(List<Pitch> list, int pc) {
397         int index = 0;
398         for (Pitch p : list) {
399             if (p.getPitchClass() == pc) {
400                 break;
401             }
402             index++;
403         }
404         return index;

500     public static int romanToArabic(String roman) {
501         int arabic = 0;
502         int max = 0;
503         char[] charArray = roman.toUpperCase(Locale.ENGLISH).toCharArray();
504         // loop from the end
505         for (int index = charArray.length - 1; index >= 0; index--) {
506             for (String c : romanSymbolMap.keySet()) {
507                 int value = romanSymbolMap.get(c).intValue();
508                 if (c.length() == 1 && c.charAt(0) == charArray[index]) {
509                     if (value >= max) {
510                         max = value;
511                         arabic += max;
512                     } else {
513                         arabic -= value;
514                     }
515                 }
516             }
517         }
518         return arabic;
519     }

521     public static String arabicToRoman(int arabic) {
522         if (logger.isDebugEnabled()) {
523             String s = String.format("converting %d", arabic);
524             logger.debug(s);
525         }
526         StringBuffer result = new StringBuffer(10);
527         for (String c : romanSymbolMap.keySet()) {
528             int value = romanSymbolMap.get(c).intValue();
529             // System.out.println("value: " + value);
530             // System.out.println("c " + c);
531             while (arabic >= value && arabic > 0) {
532                 // System.out.println("loop " + arabic);
533                 arabic -= value;
534                 result.append(c);
535             }
536         }
537         return result.toString();
538     }
