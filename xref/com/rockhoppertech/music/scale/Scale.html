138         Scale result = null;
139         try {
140             // copy the bitwise primitives
141             result = (Scale) super.clone();
142             if (spelling != null) {
143                 result.spelling = spelling;
144             }
145             if (name != null) {
146                 result.name = name;
147             }
148             if (description != null) {
149                 result.description = description;
150             }
151             result.aliases.addAll(aliases);
152 
153         } catch (final CloneNotSupportedException e) {
154             logger.error(e.getLocalizedMessage(),e);
155         }
156         return result;
157     }

182     @Override
183     public boolean equals(final Object obj) {
184         if (this == obj) {
185             return true;
186         }
187         if (obj == null) {
188             return false;
189         }
190         if (this.getClass() != obj.getClass()) {
191             return false;
192         }
193         final Scale other = (Scale) obj;
194         if (aliases == null) {
195             if (other.aliases != null) {
196                 return false;
197             }
198         } else if (!aliases.equals(other.aliases)) {
199             return false;
200         }
201         if (!Arrays.equals(degrees,
202                 other.degrees)) {
203             return false;
204         }
205         if (descendingDifferent != other.descendingDifferent) {
206             return false;
207         }
208         if (!Arrays.equals(descendingIntervals,
209                 other.descendingIntervals)) {
210             return false;
211         }
212         if (description == null) {
213             if (other.description != null) {
214                 return false;
215             }
216         } else if (!description.equals(other.description)) {
217             return false;
218         }
219         if (!Arrays.equals(intervals,
220                 other.intervals)) {
221             return false;
222         }
223         if (key == null) {
224             if (other.key != null) {
225                 return false;
226             }
227         } else if (!key.equals(other.key)) {
228             return false;
229         }
230         if (name == null) {
231             if (other.name != null) {
232                 return false;
233             }
234         } else if (!name.equals(other.name)) {
235             return false;
236         }
237         if (octave != other.octave) {
238             return false;
239         }
240         if (spelling == null) {
241             if (other.spelling != null) {
242                 return false;
243             }
244         } else if (!spelling.equals(other.spelling)) {
245             return false;
246         }
247         return true;
248     }

310     public List<Pitch> getDegreesAsPitches() {
311         final Pitch kp = PitchFactory.getPitch(key);
312         final List<Pitch> pitches = new ArrayList<Pitch>();
313         // TODO fix this spelling nonsense
314         for (final int degree : degrees) {
315             final Pitch p = PitchFactory.getPitch(degree + kp.getMidiNumber());
316             if (keySignature != null) {
317                 String preferredSpelling = null;
318                 if (keySignature.getSf() < 0) {
319                     preferredSpelling = PitchFormat.getPitchString(p,
320                             false);
321                 } else if (keySignature.getSf() > 0) {
322                     preferredSpelling = PitchFormat.getPitchString(p,
323                             true);
324                 }
325                 p.setPreferredSpelling(preferredSpelling);
326             }
327 
328             pitches.add(p);
329         }
330         return pitches;
331     }

499      * @see java.lang.Object#hashCode()

501     @Override
502     public int hashCode() {
503         final int prime = 31;
504         int result = 1;
505         result = prime * result
506                 + ((aliases == null) ? 0 : aliases.hashCode());
507         result = prime * result + Arrays.hashCode(degrees);
508         result = prime * result + (descendingDifferent ? 1231 : 1237);
509         result = prime * result + Arrays.hashCode(descendingIntervals);
510         result = prime
511                 * result
512                 + ((description == null) ? 0 : description.hashCode());
513         result = prime * result + Arrays.hashCode(intervals);
514         result = prime * result
515                 + ((key == null) ? 0 : key.hashCode());
516         result = prime * result
517                 + ((name == null) ? 0 : name.hashCode());
518         result = prime * result + octave;
519         result = prime * result
520                 + ((spelling == null) ? 0 : spelling.hashCode());
521         return result;
522     }

547         // Pitch kp = PitchFactory.getPitch(key);
548         final Pitch p = PitchFactory.getPitch(pitch);
549         final List<Pitch> pits = this.getDegreesAsPitches(key);
550         String msg = null;
551         int degree = 0;
552 
553         if (pits.contains(p)) {
554             degree = pits.indexOf(p);
555             degree++; // music is not zero based
556             msg = String.format("contains at index/degree=%d",
557                     degree);
558             Scale.logger.debug(msg);
559             return degree;
560         }
561 
562         for (int i = 0; i < pits.size(); i++) {
563             final Pitch cp = pits.get(i);
564             if (cp.getMidiNumber() == p.getMidiNumber() + 1) {
565                 msg = String.format("raised at index=%d",
566                         i);
567                 Scale.logger.debug(msg);
568                 return i;
569             }
570         }
571

574         // degree = 0;
575         // break;
576         // }
577         // }
578         return degree;

730     @Override
731     public String toString() {
732         final String ls = System.getProperty("line.separator");
733         final StringBuilder sb = new StringBuilder(this.getClass().getName())
734                 .append(ls);
735         sb.append(" name=").append(name).append(ls);
736         sb.append(" description=").append(description).append(ls);
737         sb.append(" key=").append(key).append(ls);
738         sb.append(" oct=").append(octave).append(ls);
739         sb.append(" spelling=").append(spelling).append(ls);
740         sb.append(" intervals=").append(getIntervalsAsString()).append(ls);
741         sb.append(" degrees=").append(getDegreesAsString()).append(ls);
742         sb.append(" degAsPit=").append(getDegreesAsPitchString());
743         sb.append(" isdescending=").append(isDescendingDifferent())
744                 .append(ls);
745         if (isDescendingDifferent()) {
746             sb.append(" descending=").append(getDescendingIntervalsAsString())
747                     .append(ls);
748         }
749 
750         return sb.toString();
751     }

882     public boolean contains(final String key, final Chord chord) {
883         boolean result = true;
884         final int[] chordPits = chord.getPitchClasses();
885         final List<Integer> pits = this.getDegreesAsPitchClasses(key);
886         if (Scale.logger.isInfoEnabled()) {
887             Scale.logger.info(String.format("key: '%s'",
888                     key));
889             Scale.logger.info(String.format("chord pits: '%s'",
890                     ArrayUtils.toString(chordPits)));
891             Scale.logger.info(String.format("scale: '%s'",
892                     pits));
893         }
894         for (final int cp : chordPits) {
895             if (pits.contains(cp) == false) {
896                 result = false;
897                 if (Scale.logger.isInfoEnabled()) {
898                     Scale.logger.info(String
899                             .format("does not contain chord pit %d",
900                                     cp));
901                 }
902                 break;
903             }
904         }
905         if (Scale.logger.isInfoEnabled()) {
906             Scale.logger.info(String.format("returning %b",
907                     result));
908         }
909 
910         return result;
911     }
