23  import com.rockhoppertech.collections.ArrayFunctions;
24  
25  public class ArrayFunctionsExamples {
26      
27      public static void main(String[] args) {
28  
29          // ragged initialized
30          int[][] rag = { { 1, 2 }, { 3, 4, 5 }, { 6 } };
31          ArrayFunctions.printArray(rag);
32  
33          // dynamically create the rows -- here it is ragged
34          int[][] dynamic = new int[5][];
35          for (int i = 0; i < dynamic.length; i++) {
36              dynamic[i] = new int[i * 2];
37          }
38          ArrayFunctions.printArray(dynamic);
39  
40          // arrays of object references
41          ArrayFunctions[] orefs = new ArrayFunctions[5];
42  
43          // they are not pointing anywhere
44          System.out.println("printarrays orefs should be null");
45          ArrayFunctions.printArray(orefs);
46  
47          // now they are
48          for (int i = 0; i < orefs.length; i++) {
49              orefs[i] = new ArrayFunctions();
50          }
51          System.out.println("printarrays orefs should not be null");
52          ArrayFunctions.printArray(orefs);
53  
54          // anonymous arrays of each primitive type
55          ArrayFunctions.printArray(new int[] { 4, 5, 6 });
56          ArrayFunctions.printArray(new short[] { 4, 5, 6 });
57          ArrayFunctions.printArray(new long[] { 4, 5, 6 });
58          ArrayFunctions.printArray(new double[] { 4.0, 5.0, 6.0 });
59          ArrayFunctions.printArray(new float[] { 4.0f, 5.0f, 6.0f });
60          ArrayFunctions.printArray(new byte[] { 4, 5, 6 });
61          ArrayFunctions.printArray(new char[] { '4', '5', '6' });
62          ArrayFunctions.printArray(new boolean[] { true, false, true, true });
63  
64          ArrayFunctions.printArray(new int[][] { { 4, 5 }, { 6 }, { 7, 8 } });
65          ArrayFunctions.printArray(new int[][][] { { { 4, 5 }, { 1 } },
66                  { { 2 }, { 3, 4 } } });
67      }
68  }
