26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import com.rockhoppertech.music.midi.js.MIDITrack;
30  import com.rockhoppertech.music.midi.js.MIDITrackBuilder;
31  
32  /**
33   * @author <a href="mailto:gene@rockhoppertech.com">Gene De Lisa</a>
34   * 
35   */
36  public class TrackAdditions {
37      static Logger logger = LoggerFactory.getLogger(TrackAdditions.class);
38  
39      /**
40       * @param args
41       */
42      public static void main(String[] args) {
43          // append();
44          appendNIndexes();
45          // appendN();
46  
47      }
48  
49      public static void append() {
50  
51          MIDITrack track = MIDITrackBuilder.create()
52                  .name("track 1")
53                  .noteString("C D E")
54                  .sequential()
55                  .build();
56          MIDITrack track2 = MIDITrackBuilder.create()
57                  .name("track 2")
58                  .noteString("F G")
59                  .durations(.5, .25)
60                  .sequential()
61                  .build();
62  
63          track.append(track2);
64          logger.debug("appended {}", track);
65  
66      }
67  
68      public static void appendN() {
69          MIDITrack track = MIDITrackBuilder.create()
70                  .name("track x 3")
71                  .noteString("C D E")
72                  .sequential()
73                  .build();
74          track.appendNTimes(track, 3);
75          logger.debug("appended {}", track);
76      }
77  
78      public static void appendNIndexes() {
79          MIDITrack track = MIDITrackBuilder.create()
80                  .name("track n with indexes")
81                  .noteString("C D E F G A B")
82                  .sequential()
83                  .build();
84  
85          double gap = 1.5;
86          int from = 2; // start at the E
87          int to = 5; // stop with the G
88          track.append(track, gap, from, to);
89          logger.debug("appended {}", track);
90      }
