57      static void ex01() {
58          Chord chord = ChordFactory.getChordByFullSymbol("C");
59  
60          logger.debug("here is the chord\n{}", chord);
61          MIDITrack track = chord.createMIDITrack();
62          // the pitches are in octave 0. transpose it to middle c
63          PitchModifier mod = new PitchModifier(Modifier.Operation.ADD, Pitch.C5);
64          track.map(mod);
65          track.play();
66          logger.debug("here is the track\n{}", track);
67      }

76      static void majorScaleChords() {
77          Scale scale = ScaleFactory.createFromName("Major");

104     static void inversion() {
105         Chord chord = ChordFactory.getChordByFullSymbol("C");
106         logger.debug("here is the chord\n{}", chord);
107         chord.setInversion(1);
108         logger.debug("here is the chord in 1st inversion\n{}", chord);
109         logger.debug(
110                 "here is the chord voicing in 1st inversion\n{}",
111                 chord.getChordVoicing());
112         MIDITrack track = chord.createMIDITrack();
113         logger.debug("here is the track\n{}", track);
114 
115         chord.setInversion(2);
116         logger.debug("here is the chord in 2nd inversion\n{}", chord);
117         logger.debug(
118                 "here is the chord voicing in 2nd inversion\n{}",
119                 chord.getChordVoicing());
120         track = chord.createMIDITrack();
121         logger.debug("here is the track\n{}", track);
122 
123         // oops, no 3rd inversion. No problem though, You're back in root
124         // position.
125         chord.setInversion(3);
126         logger.debug("here is the chord in 3rd inversion\n{}", chord);
127         logger.debug(
128                 "here is the chord voicing in 3rd inversion\n{}",
129                 chord.getChordVoicing());
130         track = chord.createMIDITrack();
131         logger.debug("here is the track\n{}", track);
132     }

136         ChordProgression chordProgression = ChordProgression
137                 .createFromNames(input);
138         for (Chord c : chordProgression) {
139             logger.debug("chord: {}", c);
140         }
141         Chord g7 = chordProgression.get(1);
142         g7.setInversion(2);
143         logger.debug("This should be g7 in 2nd inversion {}", g7);
144         MIDITrack track = chordProgression.createMIDITrack();
145         track.play();
146         logger.debug("track:\n{}", track);

180     static void pattern() {
181         Chord chord = ChordFactory.getChordByFullSymbol("Cmaj7");
182 
183         // indexes into the chord degrees

185         int[] pattern = { 0, 1, 2, 3 };
186

188 
189         MIDITrack track = p.createTrack();
190         track.play();

205     static void arpeggiate() {
206         Chord chord = ChordFactory.getChordByFullSymbol("Cmaj7");
207         MIDITrack track = chord.createMIDITrack();
208         logger.debug("track:\n{}", track);
209 
210         List<Double> series = new ArrayList<Double>();
211         series.add(.25);
212         series.add(.5);
213         series.add(.75);        
214         series.add(1d);
215         double startBeat = 1d;
216         double duration = 4d;
217         track.map(new ArpeggiateModifier(startBeat, duration, series,
218                 NoteModifier.Operation.ADD));
219         logger.debug("arp track:\n{}", track);
220 
221     }

226     static void lvb() {
227         String input = "I V7";
228         ChordProgression chordProgression = ChordProgression
229                 .createFromRoman(input);
230         logger.debug(
231                 "chordprogression in default C major: \n{}",
232                 chordProgression);
233         for (Chord c : chordProgression) {
234             logger.debug("chord: {}", c);
235         }
236 
237         logger.debug("chord progression: \n{}", chordProgression);
238         logger.debug(
239                 "chord progression end beat: \n{}",
240                 chordProgression.getEndBeat());
241         chordProgression.append(chordProgression);
242         logger.debug("chord progression after append: \n{}", chordProgression);
243 
244         chordProgression.append(chordProgression).append(chordProgression);
245 
246         // append a C major
247         chordProgression.add(ChordFactory.getChordByFullSymbol("C"));
248 
249         logger.debug(
250                 "chord progression after more appends: \n{}",
251                 chordProgression);
252 
253         MIDITrack track = chordProgression.createMIDITrack();
254         track.play();
255         // logger.debug("track:\n{}", track);
256     }
