23  import java.io.IOException;
24  
25  import javax.sound.midi.InvalidMidiDataException;
26  import javax.sound.midi.Sequence;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import com.rockhoppertech.music.Duration;
32  import com.rockhoppertech.music.midi.js.Instrument;
33  import com.rockhoppertech.music.midi.js.KeySignature;
34  import com.rockhoppertech.music.midi.js.MIDIPerformer;
35  import com.rockhoppertech.music.midi.js.MIDITrack;
36  import com.rockhoppertech.music.midi.js.MIDITrackBuilder;
37  import com.rockhoppertech.music.midi.js.Score;
38  import com.rockhoppertech.music.midi.js.ScoreFactory;
39  
40  public class StringQuartet {
41  
42      static String filename = "src/test/resources/midifiles/stringquartet.mid";
43      private static final Logger logger = LoggerFactory
44              .getLogger(StringQuartet.class);
45  
46      public static void main(String[] args) {
47  
48          createQuartet();
49          readAndPlay();
50      }
51  
52      static void readAndPlay() {
53          Score score = null;
54          try {
55              score = ScoreFactory.readSequence(filename);
56          } catch (InvalidMidiDataException | IOException e) {
57              e.printStackTrace();
58              return;
59          }
60          logger.debug("Read score {}", score);
61  
62          Sequence sequence = ScoreFactory.scoreToSequence(score);
63          if (sequence == null) {
64              logger.error("oops. cannot create Sequence");
65              return;
66          }
67          MIDIPerformer perf = new MIDIPerformer();
68          perf.score(score)
69                  .atTempo(120)
70                  .play();
71      }
72  
73      static void createQuartet() {
74          Score score = new Score("String Quartet");
75          score.setTempoAtBeat(1d, 120);
76          score.setKeySignatureAtBeat(1d, KeySignature.DMAJOR);
77          score.setTimeSignatureAtBeat(1d, 3, 4);
78          score.setTempoAtBeat(4d, 240);
79          score.setKeySignatureAtBeat(4d, KeySignature.BFMINOR);
80          score.setTimeSignatureAtBeat(4d, 5, 4);
81  
82          MIDITrack violin = MIDITrackBuilder.create()
83                  .name("Violin I")
84                  .instrument(Instrument.VIOLIN)
85                  .noteString("S+ C6 D E")
86                  .build();
87          score.add(violin);
88  
89          MIDITrack violin2 = MIDITrackBuilder.create()
90                  .name("Violin II")
91                  .instrument(Instrument.VIOLIN)
92                  .noteString("S+ C5 D E")
93                  .build();
94          score.add(violin2);
95  
96          MIDITrack repeatedLick = MIDITrackBuilder.create()
97                  .instrument(Instrument.VIOLIN)
98                  .noteString("S+ C5,e D,e E,e A,e")
99                  .repeats(10)
100                 .build();
101         violin2.append(repeatedLick);
102 
103         MIDITrack viola = MIDITrackBuilder.create()
104                 .name("Viola")
105                 .instrument(Instrument.VIOLA)
106                 .noteString("S+ C4 D E")
107                 .build();
108         score.add(viola);
109 
110         MIDITrack contrabass = MIDITrackBuilder.create()
111                 // .name("Bass")
112                 .instrument(Instrument.CONTRABASS)
113                 .noteString("S+ E2 F E ")
114                 .build();
115         score.add(contrabass);
116 
117         MIDITrack tmp = MIDITrackBuilder.create()
118                 .noteString("S+ X8 (G2,.5)")
119                 .instrument(Instrument.CONTRABASS)
120                 // otherwise, it defaults to Piano, and is then remapped to
121                 // contrabass.
122                 .build();
123         contrabass.append(tmp).append(tmp);
124 
125         tmp = MIDITrackBuilder.create()
126                 .noteString("S+ X8 (G4,.5, A4,.5)")
127                 .instrument(Instrument.VIOLA)
128                 .build();
129         viola.append(tmp, Duration.DOUBLE_WHOLE_NOTE);
130 
131         logger.debug("Generated score {}", score);
132 
133         ScoreFactory.writeToMIDIFile(filename, score);
134     }
