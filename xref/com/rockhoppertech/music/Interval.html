191     /**
192      * Turn an array into a String.
193      * 
194      * @param a an array
195      * @return a {@code String} with the array contents
196      */
197     public static String arrayToString(final int[] a) {
198         final StringBuilder sb = new StringBuilder();
199         for (int j = 0; j < a.length; j++) {
200             sb.append(a[j]);
201             if (j < a.length - 1) {
202                 sb.append(',');
203             }
204         }
205         return sb.toString();
206     }

208     /**
209      * 
210      * @param spelling
211      *            is a chord spelling
212      * @return an array of intervals
213      */
214     public static int[] degreesToIntervals(final String spelling) {
215         Interval.logger.debug("spelling: \'" + spelling + "\'");
216         // \s is A whitespace character: [ \t\n\x0B\f\r]
217         // see java.util.regex.Pattern
218         final String[] degrees = spelling.split("\\s");
219         for (final String degree : degrees) {
220             Interval.logger.debug("degrees: " + degree);
221         }
222         // given "1 3 5" split will give an array
223         // with 1 in index 0 3 in index 2 etc
224 
225         final int[] intervals = new int[degrees.length - 1];
226         // all of the degrees start with 1 so we skip that one
227         // for 1 3 5 we get 4 7 for the intervals
228         for (int i = 0; i < intervals.length; i++) {
229             intervals[i] = Interval.getSpelling(degrees[i + 1]);
230             Interval.logger.debug("intervals {}", intervals[i]);
231         }
232         return intervals;
233     }

235     /**
236      * input can be comma or whitespace delimited.
237      * 
238      * @param s
239      *            a scale spelling
240      * @return an array of intervals
241      */
242     public static int[] getIntervalsFromDegreeString(final String s) {
243 
244         final Scanner scanner = new Scanner(s);
245         if (s.indexOf(',') != -1) {
246             scanner.useDelimiter(",");
247             Interval.logger.debug("comma delimited");
248         }
249 
250         final List<Integer> pitches = new ArrayList<Integer>();
251         while (scanner.hasNext()) {
252             final String string = scanner.next().trim();
253             Interval.logger.debug("degreestring: " + string);
254 
255             int pc = Interval.getScaleSpelling(string);
256             pc++;
257             pitches.add(pc);
258         }
259         final int size = pitches.size();
260         final int[] array = new int[size - 1];
261         for (int i = 1; i < size; i++) {
262             final int val = pitches.get(i) - pitches.get(i - 1);
263             Interval.logger.debug("val {}", val);
264             array[i - 1] = val;
265         }
266         scanner.close();
267         return array;
268     }

401     /**
402      * Return intervals in relation to first int. Can be comma delimted or not.
403      * Non specific: can be ints, midinumbers etc.
404      * 
405      * @param s
406      *            a string that can be parsed into an int
407      * @return an int[] array of absolute intervals
408      */
409     public static int[] getAbsoluteIntervalsFromString(final String s) {
410 
411         final Scanner scanner = new Scanner(s);
412         if (s.indexOf(',') != -1) {
413             scanner.useDelimiter(",");
414         }
415 
416         final List<Integer> ints = new ArrayList<Integer>();
417         while (scanner.hasNext()) {
418             final String interval = scanner.next().trim();
419             ints.add(Integer.parseInt(interval));
420         }
421         final int size = ints.size();
422         final int[] array = new int[size - 1];
423         int root = ints.get(0);
424         for (int i = 1; i < size; i++) {
425             final int val = ints.get(i) - root;
426             array[i - 1] = val;
427         }
428         scanner.close();
429 
430         return array;
431     }

490     /**
491      * Turn intervals into Scale degrees.
492      * 
493      * 
494      * @see com.rockhoppertech.music.scale.Scale
495      * @param intervals
496      *            the intervals to turn into degrees
497      * @return an array of degrees
498      */
499     public static int[] intervalsToDegrees(final int[] intervals) {
500         final int[] degrees = new int[intervals.length + 1];
501         // degrees[0] = 1;
502         for (int i = 0; i < intervals.length; i++) {
503             degrees[i + 1] = degrees[i] + intervals[i];
504         }
505         return degrees;
506     }

508     /**
509      * Can turn the intervals in a scale into their scale degrees. You can
510      * generate patterns.
511      * 
512      * @see com.rockhoppertech.music.scale.ScaleFactory#getNoteListPattern(Scale, int[], int, int, int, double,
513      *      boolean, double, boolean)
514      * @param intervals
515      *            an array of intervals
516      * @param nOctaves
517      *            how many octaves to use
518      * @return an array of degrees
519      */
520     public static int[] intervalsToDegrees(final int[] intervals,
521             final int nOctaves) {
522         final int[] roots = { 0, 12, 24, 36, 48, 60, 72, 84, 96, 108, 120 };
523         final List<Integer> list = new ArrayList<Integer>();
524         list.add(0);
525         for (int octave = 0; octave < nOctaves; octave++) {
526             int previous = roots[octave];
527             Interval.logger.debug("p " + previous);
528             for (final int interval : intervals) {
529                 final int n = previous + interval;
530                 list.add(n);
531                 previous = n;
532                 Interval.logger.debug("pp " + previous);
533                 Interval.logger.debug("int " + interval);
534             }
535         }
536 
537         Integer[] ia = new Integer[list.size()];
538         ia = list.toArray(ia);
539         final int[] degrees = new int[ia.length];
540         for (int i = 0; i < ia.length; i++) {
541             Interval.logger.debug("degree " + ia[i]);
542             degrees[i] = ia[i].intValue();
543         }
544         return degrees;
545     }
