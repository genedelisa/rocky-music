165     /**
166      * Copies new events into the TimeSeries parameter - which is also returned.
167      * 
168      * @param ts
169      *            a {@code TimeSeries} that will be added to
170      * @param s
171      *            a duration string
172      * @return the same{@code TimeSeries} that you passed in
173      */
174     public static TimeSeries getDurationAsTimeSeries(TimeSeries ts, String s) {
175         String token = null;
176         Scanner scanner = new Scanner(s);
177 
178         if (s.indexOf(',') != -1) {
179             scanner.useDelimiter(",");
180         }
181 
182         while (scanner.hasNext()) {
183             if (scanner.hasNext(dpattern)) {
184                 token = scanner.next(dpattern);
185                 double d = getDottedValue(token);
186                 ts.add(d);
187                 logger.debug("'{}' is dotted value is {}",
188                         token, d);
189 
190             } else if (scanner.hasNext(pattern)) {
191                 token = scanner.next(pattern);
192                 double d = durKeyMap.get(token);
193                 ts.add(d);
194                 logger.debug("'{}' is not dotted value is {}", token, d);
195                 // } else if (scanner.hasNext(tripletPattern)) {
196                 // token = scanner.next(tripletPattern);
197                 // double d = durKeyMap.get(token);
198                 // ts.add(d);
199                 // System.out.println(String
200                 // .format("'%s' is not dotted value is %f",
201                 // token,
202                 // d));
203             } else if (scanner.hasNextDouble()) {
204                 double d = scanner.nextDouble();
205                 ts.add(d);
206                 logger.debug("{} is a double", d);
207             } else {
208                 // just ignore it. or throw exception?
209                 String skipped = scanner.next();
210                 logger.debug("skipped '{}'", skipped);
211             }
212         }
213         scanner.close();
214         return ts;
215     }

217     private static double getDottedValue(String token) {
218         String s = token.substring(0, token.indexOf('.'));
219         double val = durKeyMap.get(s);
220         int dots = StringUtils.countMatches(token, ".");
221         val = Duration.getDotted(val, dots);
222         return val;
223     }

225     private static String getKeyString(double duration) {
226         for (Entry<String, Double> e : durKeyMap.entrySet()) {
227             if (Math.abs(duration - e.getValue()) < .0000001) {
228                 // if (e.getValue() == duration) {
229                 return e.getKey();
230             }
231         }
232         return null;
233     }

296     public static String getDurationString(double d) {
297         StringBuilder sb = new StringBuilder();
298         boolean foundInMap = false;
299         if (durKeyMap.containsValue(d)) {
300             sb.append(getKeyString(d)).append(' ');
301             // System.err.printf("contains value %s %f\n", sb, d);
302             foundInMap = true;
303         }
304 
305         String nearest = getNearestKeyString(d);
306         double nearestDuration = durKeyMap.get(nearest);
307         // System.err.printf("%fs nearest value %s nd %f\n", d, nearest,
308         // nearestDuration);
309 
310         boolean foundDotted = false;
311         for (int i = 1; i < 3; i++) {
312             double dotted = Duration.getDotted(nearestDuration, i);
313             // System.err.printf("dotted %d %f = %f?\n", i, dotted, d);
314             // if (d == dotted) {
315             if (Math.abs(d - dotted) < .0000001) {
316 
317                 foundDotted = true;
318                 sb.append(nearest);
319                 for (int j = 0; j < i; j++) {
320                     sb.append('.');
321                 }
322                 sb.append(' ');
323             }
324         }
325 
326         // not in the map and not dotted, just give the value
327         if (foundDotted == false && foundInMap == false) {
328             sb.append(d).append(' ');
329         }
330 
331         return sb.toString().trim();
