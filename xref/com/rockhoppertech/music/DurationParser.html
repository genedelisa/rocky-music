87      public static double getDuration(String durationString) {
88          // is it dotted?
89          // Pattern dot = Pattern.compile("\\.+");
90          if (dpattern.matcher(durationString).matches()) {
91              // Matcher m = dpattern.matcher(durationString);
92              // String token = durationString.substring(m.start(),
93              // m.end());
94              return getDottedValue(durationString);
95          }
96          return durKeyMap.get(durationString);
97      }

177     private static double getDottedValue(String token) {
178         String s = token.substring(0, token.indexOf('.'));
179         double val = durKeyMap.get(s);
180         int dots = StringUtils.countMatches(token, ".");
181         val = Duration.getDotted(val, dots);
182         return val;
183     }

185     private static String getKeyString(double duration) {
186         for (Entry<String, Double> e : durKeyMap.entrySet()) {
187             if (Math.abs(duration - e.getValue()) < .0000001) {            
188             //if (e.getValue() == duration) {
189                 return e.getKey();
190             }
191         }
192         return null;
193     }

256     public static String getDurationString(double d) {
257         StringBuilder sb = new StringBuilder();
258         boolean foundInMap = false;
259         if (durKeyMap.containsValue(d)) {
260             sb.append(getKeyString(d)).append(' ');
261             // System.err.printf("contains value %s %f\n", sb, d);
262             foundInMap = true;
263         }
264 
265         String nearest = getNearestKeyString(d);
266         double nearestDuration = durKeyMap.get(nearest);
267         // System.err.printf("%fs nearest value %s nd %f\n", d, nearest,
268         // nearestDuration);
269 
270         boolean foundDotted = false;
271         for (int i = 1; i < 3; i++) {
272             double dotted = Duration.getDotted(nearestDuration, i);
273             // System.err.printf("dotted %d %f = %f?\n", i, dotted, d);
274             // if (d == dotted) {
275             if (Math.abs(d - dotted) < .0000001) {
276 
277                 foundDotted = true;
278                 sb.append(nearest);
279                 for (int j = 0; j < i; j++) {
280                     sb.append('.');
281                 }
282                 sb.append(' ');
283             }
284         }
285 
286         // not in the map and not dotted, just give the value
287         if (foundDotted == false && foundInMap == false) {
288             sb.append(d).append(' ');
289         }
290 
291         return sb.toString().trim();
