617     @Test
618     public void createFromFrequency() {
619         for (double fq : PitchFactory.EQUAL_TEMPERAMENT.values()) {
620             logger.debug("et fq {}", fq);
621         }
622 
623         double freq = 261.6255653005986;
624         Pitch p = PitchFactory.createFromFrequency(freq);
625         Pitch p2 = PitchFactory.createFromFrequency(freq);
626         logger.debug(p.toProlixString());
627         
628         assertThat("The pitches are the same",
629                    p,
630                    sameInstance(p2));
631         assertThat("The pitch is not null",
632                    p,
633                    notNullValue());
634         assertThat("The fq is correct",
635                    p.getFrequency(),
636                    equalTo(freq));
637         assertThat("The midi number is correct",
638                    p.getMidiNumber(),
639                    equalTo(60));
640         assertThat("The pitch is middle c",
641                    p,
642                    equalTo(new Pitch(Pitch.C5)));
643         assertThat("The cents is correct",
644                    p.getCents(),
645                    equalTo((short) 0));
646         assertThat("The pitchbend is correct",
647                    p.getPitchBend(),
648                    equalTo((short) 0));
649 
650         freq = 261;
651         p = PitchFactory.createFromFrequency(freq);
652         p2 = PitchFactory.createFromFrequency(freq);
653         assertThat("The pitches are the same",
654                    p,
655                    sameInstance(p2));
656         logger.debug(p.toProlixString());
657         assertThat("The pitch is not null",
658                    p,
659                    notNullValue());
660         assertThat("The fq is correct",
661                    p.getFrequency(),
662                    equalTo(freq));
663         assertThat("The midi number is correct",
664                    p.getMidiNumber(),
665                    equalTo(60));
666         assertThat("The cents is correct",
667                    p.getCents(),
668                    equalTo((short) -4));
669         assertThat("The pitchbend is correct",
670                    p.getPitchBend(),
671                    equalTo((short) 8188));
