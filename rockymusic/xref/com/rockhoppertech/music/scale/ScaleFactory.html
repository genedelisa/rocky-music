286     public static MIDITrack createMIDITrack(final int[] intervals,
287             int rootMidiNum, double startBeat, final double duration,
288             final int nOctaves, final boolean down) {
289 
290         logger.debug("root " + rootMidiNum);
291 
292         final MIDITrack notelist = new MIDITrack();
293 
294         MIDINote note = new MIDINote(rootMidiNum, startBeat, duration);
295         notelist.add(note);
296 
297         int previous = rootMidiNum;
298         for (int oct = 0; oct < nOctaves; oct++) {
299             for (final int interval : intervals) {
300                 // FIXME
301                 if (down) {
302                     note = new MIDINote(previous - interval,
303                             startBeat += duration, duration);
304                 } else {
305                     note = new MIDINote(previous + interval,
306                             startBeat += duration, duration);
307                 }
308                 notelist.add(note);
309                 previous = note.getMidiNumber();
310             }
311             if (down) {
312                 rootMidiNum -= 12;
313             } else {
314                 rootMidiNum += 12;
315             }
316 
317             previous = rootMidiNum;
318         }
319         return notelist;
320     }

342             final double startBeat, final double duration) {
343 
344         final MIDITrack track = new MIDITrack();
345         track.setName(scale.getName());
346         String s = String.format(
347                 "created track from scale %s, start beat %f duration %f",
348                 scale.getName(),
349                 startBeat,
350                 duration);
351         track.addMetaText(1d, s);
352 
353         final int rootMidiNum = PitchFactory.getPitch(
354                 scale.getKey() + scale.getOctave()).getMidiNumber();
355         MIDINote note = new MIDINote(rootMidiNum, startBeat, duration);
356         track.add(note);
357         final int[] intervals = scale.getIntervals();
358         int previous = rootMidiNum;
359         for (final int interval : intervals) {
360             note = new MIDINote(previous + interval, startBeat, duration);
361             track.add(note);
362             previous = note.getMidiNumber();
363         }
364         return track;
365     }

421         track.add(note);

425             note = new MIDINote(previous + interval, startBeat += duration,
426                     duration);
427             track.add(note);

434     public static MIDITrack createMIDITrack(final Scale scale,
435             final int rootMidiNum, final double startBeat,
436             final double duration, final boolean upAndDown, final int nOct) {
437 
438         final MIDITrack notelist = ScaleFactory.createMIDITrack(scale,
439                 rootMidiNum, startBeat, duration, nOct);
440 
441         if (upAndDown) {
442             if (scale.isDescendingDifferent()) {
443                 final MIDITrack reverse = ScaleFactory.createMIDITrack(
444                         scale.getDescendingIntervals(), rootMidiNum + 12, 1d,
445                         duration, nOct, true);
446                 notelist.append(reverse);
447                 notelist.sequential();
448             } else {
449                 final MIDITrack reverse = notelist.retrograde();
450                 notelist.append(reverse);
451                 notelist.sequential();
452             }
453         }
454         return notelist;
455     }

625     public static List<Pitch> getPitches(final String name,
626             final int rootMidiNum) {
627         if (scaleMap == null) {
628             ScaleFactory.init();
629         }
630         final List<Pitch> pitches = new ArrayList<Pitch>();
631         final Scale scale = ScaleFactory.createFromName(name);
632         Pitch pitch = PitchFactory.getPitch(rootMidiNum);
633         pitches.add(pitch);
634         final int[] intervals = scale.getIntervals();
635         int previous = rootMidiNum;
636         for (final int interval : intervals) {
637             pitch = PitchFactory.getPitch(previous + interval);
638             pitches.add(pitch);
639             previous = pitch.getMidiNumber();
640         }
641         return pitches;
642     }

914     /**
915      * E.g. for a c major scale, mode 1 will be d dorian. The returned scale
916      * will have the same octave as the one passed in. The key will of course be
917      * different (e.g. for mode 1 in c major it will be d)
918      * 
919      * @param scale
920      * @param mode
921      * @return the mode for the scale
922      */
923     public static Scale getMode(final Scale scale, int mode) {
924         List<Pitch> degrees = scale.getDegreesAsPitches();
925         Pitch key = degrees.get(mode);
926 
927         // see how rotate works. for mode 1 you'd have to pass in -1 otherwise.
928         // not especially intuitive from the caller's pov.
929         mode *= -1;
930 
931         int[] i = scale.getIntervals();
932         List<Integer> intervals = new ArrayList<Integer>();
933         for (int element : i) {
934             intervals.add(element);
935         }
936         Collections.rotate(intervals, mode);
937         i = new int[intervals.size()];
938         int j = 0;
939         for (Integer in : intervals) {
940             i[j++] = in;
941         }
942 
943         String name = String.format("%s mode %d", scale.getName(), mode);
944         Scale scaleMode = new Scale(name, i);
945         scaleMode.setKey(PitchFormat.getPitchString(key));
946         scaleMode.setOctave(scale.getOctave());
947 
948         logger.debug("scale {}", scale);
949         logger.debug("scale mode {}", scaleMode);
950 
951         return scaleMode;
952     }
