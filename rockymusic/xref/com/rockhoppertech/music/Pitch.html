37  /**
38   * <p>
39   * A representation of musical frequency (Pitch).
40   * <p>
41   * <p>
42   * Internally represented by a midi number. The frequency is calculated based on
43   * the MIDI number.
44   * </p>
45   * 
46   * See http://en.wikipedia.org/wiki/Scientific_pitch_notation
47   * 
48   * For MusicXML, Common Lisp Music and others the octave is "off" by 1. In MIDI
49   * middle c is C5, for them it is C4.
50   * 
51   * <p>
52   * For equal temperament you should use the PitchFactory to obtain instances a
53   * la the GoF Flyweight design pattern. (Most likely you will have zillions of
54   * instances of Pitch in your program. The Flyweight helps this problem).
55   * </p>
56   * 
57   * 
58   * @author <a href="mailto:gene@rockhoppertech.com">Gene De Lisa</a>
59   * @version 1.0 Mon Aug 10, 1998
60   * @see com.rockhoppertech.music.PitchFactory
61   * @see com.rockhoppertech.music.PitchFormat
62   */

109     /**
110      * <p>
111      * Use the PitchFactory to get an instance instead. Use this only in special
112      * circumstances
113      * </p>
114      * <p>
115      * Can use a constant.
116      * <p>
117      * 
118      * {@code
119      * new Pitch(Pitch.C5);
120 	 * }
121      * 
122      * @param n
123      *            the midi number to use. 60 is middle c.
124      */
125     public Pitch(int n) {
126         if (n < 0 || n > 127) {
127             throw new IllegalArgumentException("MIDI number out of range " + n);
128         }
129         midiNumber = n;
130         // this.frequency = Pitch.midiFq(this.midiNumber);
131         frequency = PitchFactory.EQUAL_TEMPERAMENT.get(midiNumber);
132     }

184      * seconds (half steps). Can use constants.
185      * </p>
186      * <p>
187      * This uses the PitchFactory to return the appropriate instance. The object
188      * that is the receiver of this message is not affected in any way.

192      * p.transpose(Pitch.MAJOR_SEVENTH);

194      * 
195      * @param minorSeconds
196      *            The value added with the receiver's midinumber. Can be
197      *            negative to transpose down.
198      * @return the transposed Pitch
199      */
200     public Pitch transpose(int minorSeconds) {
201         int n = midiNumber + minorSeconds;
202         Pitch p = PitchFactory.getPitch(n);
203         logger.debug(String.format("transposed to %d", n));
204         return p;
205     }

233     /**
234      * Check to see if the pitch is diatonic or not.
235      * 
236      * @param num
237      *            The midi number
238      * @return whether it is an accidental
239      */
240     public static boolean isAccidental(int num) {
241         boolean r = false;
242         switch (num % 12) {
243         case 1:
244         case 3:
245         case 6:
246         case 8:
247         case 10:
248             r = true;
249             break;
250         default:
251             r = false;
252             break;
253         }
254         return r;
255     }

257     public final static int REGEX_FLAGS = Pattern.CASE_INSENSITIVE
258             | Pattern.UNICODE_CASE | Pattern.CANON_EQ;
259 
260     public static boolean isSharp(String pitchName) {
261         String sharpNoteNameRegexp = "([A-G])(#|s|x)";
262         Pattern pattern = Pattern.compile(sharpNoteNameRegexp,
263                 REGEX_FLAGS);
264         Matcher match = pattern.matcher(pitchName);
265         if (match.matches()) {
266             logger.debug("{} matches", pitchName);
267             return true;
268         }
269         return false;
270     }
271 
272     public static boolean isFlat(String pitchName) {
273         String flatNoteNameRegexp = "([A-G])(-|b|f)";
274         Pattern pattern = Pattern.compile(flatNoteNameRegexp,
275                 REGEX_FLAGS);
276         Matcher match = pattern.matcher(pitchName);
277         if (match.matches()) {
278             logger.debug("{} matches", pitchName);
279             return true;
280         }
281         return false;
282     }

284     /**
285      * <p>
286      * Calculate frequency of a pitch class/octave pair.
287      * </p>
288      * 
289      * @param pitch
290      *            the pitch class
291      * @param oct
292      *            the octave. 6.0 is middle C's octave
293      * @return the frequency
294      */
295     public static double midiFq(double pitch, double oct) {
296         return (REF_PITCH * Math.pow(2d,
297                 (oct - 5d)) * Math.pow(2d,
298                 (pitch / 12d)));
299     }
300 
301     /**
302      * Calculate equal tempered frequency of a midi note. The PitchFactory
303      * contains a cached table of Equal Temp. pitches. so use that instead of
304      * this method.
305      * 
306      * Called by {@code PitchFactory} when calculating {@code EQUAL_TEMPERAMENT}
307      * .
308      * 
309      * <pre>
310      * {@code
311 	 * frequency = PitchFactory.EQUAL_TEMPERAMENT.get(this.midiNumber);
312 	 * }
313 	 * </pre>
314      * 
315      * @param midiNoteNumber
316      *            an <code>int</code> value
317      * @return the frequency
318      */
319     public static double midiFq(int midiNoteNumber) {
320         int pitch = midiNoteNumber / 12;
321         int oct = midiNoteNumber % 12;
322         double dp = pitch - 5d;
323         double doct = oct / 12d;
324         return (REF_PITCH * Math.pow(2d,
325                 dp) * Math.pow(2d,
326                 doct));
327     }

329     /*
330      * <p>Uses the MIDI number for natural ordering. </p>
331      * 
332      * @see java.lang.Comparable#compareTo(java.lang.Object)
333      */
334     @Override
335     public int compareTo(Pitch other) {
336 
337         int val = 0;
338         if (getMidiNumber() == other.getMidiNumber()) {
339             val = 0;
340         } else if (getMidiNumber() < other.getMidiNumber()) {
341             val = -1;
342         } else {
343             val = 1;
344         }
345         return val;
346     }

796      */
797     @Override
798     public int hashCode() {
799         final int PRIME = 31;
800         int result = 1;
801         result = PRIME * result + midiNumber;
802         return result;
803     }

805     /*
806      * <p>Compares just MIDI number. </p>
807      * 
808      * @see java.lang.Object#equals(java.lang.Object)
809      */
810     @Override
811     public boolean equals(Object obj) {
812         if (this == obj) {
813             return true;
814         }
815         if (obj == null) {
816             return false;
817         }
818         if (getClass() != obj.getClass()) {
819             return false;
820         }
821         final Pitch other = (Pitch) obj;
822         if (midiNumber != other.midiNumber) {
823             return false;
824         }
825         return true;
826     }

879     public static Set<String> getSynonyms(int midiNumber) {
880         Set<String> syn = new TreeSet<String>();
881         for (Map.Entry<String, Integer> entry : nameMap.entrySet()) {
882             if (entry.getValue() == midiNumber) {
883                 // don't really feel like writing custom JUnit matchers
884                 String key = entry.getKey();
885                 syn.add(key);
886                 syn.add(key.toLowerCase(Locale.ENGLISH));
887                 syn.add(key.toUpperCase(Locale.ENGLISH));
888                 syn.add(StringUtils.capitalize(key.toLowerCase(Locale.ENGLISH)));
889             }
890         }
891         return syn;
892     }
