176     /**
177      * The intervals are in relation to the root. So "1 3 5" returns "4 7" and
178      * not "4 3". <blockquote>
179      * 
180      * <pre>
181      * int[] intervals = ChordFactory.degreesToIntervals(&quot;1 3 5&quot;);
182      * </pre>
183      * 
184      * </blockquote>
185      * 
186      * @param spelling
187      *            A chord spelling as a String
188      * @return The intervals as an int[]
189      * 
190      * @see com.rockhoppertech.music.Interval
191      */
192     public static int[] degreesToIntervals(String spelling) {
193         logger.debug("spelling: \'" + spelling + "\'");
194         // \s is A whitespace character: [ \t\n\x0B\f\r]
195         // see java.util.regex.Pattern
196         String[] degrees = spelling.split("\\s");
197         for (int i = 0; i < degrees.length; i++) {
198             logger.debug("degrees: " + degrees[i]);
199         }
200         // given "1 3 5" split will give an array
201         // with 1 in index 0 3 in index 2 etc
202 
203         int[] intervals = new int[degrees.length - 1];
204         // all of the degrees start with 1 so we skip that one
205         // for 1 3 5 we get 4 7 for the intervals
206         for (int i = 0; i < intervals.length; i++) {
207             intervals[i] = Interval.getSpelling(degrees[i + 1]);
208             logger.debug("{}", intervals[i]);
209         }
210         return intervals;
211     }

333                 + ArrayUtils.toString(someIntervals));

404     /**
405      * this is called from Chord when you change its instance values. So
406      * chord.getNotelist() returns the result.
407      * 
408      * @param chord
409      *            the chord instance to use
410      * @return a {@cod MIDITrack} instance
411      * 
412      * @see Chord#createMIDITrack()
413      */
414     public static MIDITrack createMIDITrack(Chord chord) {
415         MIDITrack notelist = new MIDITrack();
416         int[] intervals = chord.getIntervals();
417         int drop = chord.getDrop();
418         if (chord.getInversion() == 0) {
419             MIDINote note = new MIDINote(chord.getRoot(), chord.getStartBeat(),
420                     chord.getDuration());
421             notelist.add(note);
422 
423             for (int i = 0; i < intervals.length; i++) {
424                 note = new MIDINote(chord.getRoot() + intervals[i],
425                         chord.getStartBeat(), chord.getDuration());
426                 notelist.add(note);
427             }
428         } else {
429             StringBuilder sb = new StringBuilder();
430             int index = chord.getInversion() - 1;
431             int leap = 12;
432             int max = chord.getMaxInterval();
433             if (logger.isDebugEnabled()) {
434                 logger.debug("max interval " + max);
435             }
436             while (max > leap)
437                 leap += 12; // 9th, 11th, 13th chords
438 
439             // sb.append(Chord.intervalToChordDegree(leap)).append(" ");
440             MIDINote note = new MIDINote(chord.getRoot() + leap,
441                     chord.getStartBeat(), chord.getDuration());
442             notelist.add(note);
443 
444             for (int i = 0; i < index && i < intervals.length; i++) {
445                 // sb.append(Chord.intervalToChordDegree(intervals[i] +
446                 // leap)).append(" ");
447                 note = new MIDINote(chord.getRoot() + intervals[i] + leap,
448                         chord.getStartBeat(), chord.getDuration());
449                 notelist.add(note);
450             }
451 
452             for (int i = index; i < intervals.length; i++) {
453                 // sb.append(Chord.intervalToChordDegree(intervals[i] + +
454                 // intervals[i])).append(" ");
455                 note = new MIDINote(chord.getRoot() + intervals[i],
456                         chord.getStartBeat(), chord.getDuration());
457                 notelist.add(note);
458             }
459 
460             notelist.sortByAscendingPitches();
461             logger.debug("{}", notelist);
462             int p = chord.getRoot() % 12;
463             int o = chord.getRoot() / 12;
464             for (MIDINote n : notelist) {
465                 int oo = n.getMidiNumber() / 12;
466                 int nPluses = oo - o;
467                 int in = n.getMidiNumber() % 12;
468                 in = Math.abs(in - p);
469                 logger.debug(String.format("in %d ", in));
470                 for (int i = 0; i < nPluses; i++)
471                     sb.append('+');
472                 sb.append(Chord.intervalToChordDegree(in)).append(' ');
473             }
474             // ChordVoicing chordVoicing = new ChordVoicing(chord.getRoot() /
475             // 12,
476             // sb.toString());
477             chord.setChordVoicing(sb.toString());
478             if (logger.isDebugEnabled()) {
479                 String s = String.format("inversion %d is this voicing %s %s",
480                         chord.getInversion(), sb.toString(), sb.toString());
481                 logger.debug(s);
482             }
483         }
484 
485         if (drop > 0) {
486             notelist.sortByAscendingPitches();
487             int size = notelist.size();
488             MIDINote n = notelist.get(size - drop);
489             logger.debug("Dropping " + n);
490             if (n.getMidiNumber() >= 12) {
491                 n.setPitch(n.getPitch().transpose(-12));
492                 logger.debug("Dropped " + n);
493             } else {
494                 System.err.println("Too low to drop " + n);
495             }
496         }
497         return notelist;
498     }

500     public static MIDITrack createMIDITrack(int rootMidiNum, Chord chord,
501             double startBeat, double duration, int inversion, int drop) {
502 
503         MIDITrack notelist = new MIDITrack();
504         int[] intervals = chord.getIntervals();
505         if (inversion > intervals.length) {
506             throw new IllegalArgumentException("Inversion too large "
507                     + inversion);
508         }
509 
510         if (inversion == 0) {
511             MIDINote note = new MIDINote(rootMidiNum, startBeat, duration);
512             notelist.add(note);
513 
514             for (int i = 0; i < intervals.length; i++) {
515                 note = new MIDINote(rootMidiNum + intervals[i], startBeat,
516                         duration);
517                 notelist.add(note);
518             }
519         } else {
520             int index = inversion - 1;
521             MIDINote note = new MIDINote(rootMidiNum + 12, startBeat, duration);
522             notelist.add(note);
523 
524             for (int i = 0; i < index && i < intervals.length; i++) {
525                 note = new MIDINote(rootMidiNum + intervals[i] + 12, startBeat,
526                         duration);
527                 notelist.add(note);
528             }
529 
530             for (int i = index; i < intervals.length; i++) {
531                 note = new MIDINote(rootMidiNum + intervals[i], startBeat,
532                         duration);
533                 notelist.add(note);
534             }
535         }
536 
537         if (drop > 0) {
538             // sort by pitch
539             Comparator<MIDINote> comp = new Comparator<MIDINote>() {
540                 public int compare(MIDINote o1, MIDINote o2) {
541                     if (o1.getMidiNumber() > o2.getMidiNumber()) {
542                         return 1;
543                     } else if (o1.getMidiNumber() < o2.getMidiNumber()) {
544                         return -1;
545                     }
546                     return 0;
547                 }
548             };
549             notelist.sort(comp);
550             int size = notelist.size();
551             MIDINote n = notelist.get(size - drop);
552             logger.debug("Dropping " + n);
553             n.setPitch(n.getPitch().transpose(-12));
554             logger.debug("Dropped " + n);
555             // MIDINote h = notelist.getHighestPitchedNote();
556         }
557         return notelist;
558     }
