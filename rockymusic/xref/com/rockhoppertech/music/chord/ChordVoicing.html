57      public MIDITrack getNoteList(int octave, Chord c) {
58          Chord cl = (Chord) c.clone();
59          int root = c.getRoot();
60          int pc = root % 12;
61          int n = octave * 12 + pc;
62          cl.setRoot(n);
63          return parse(this.voicing, cl);

102     /**
103      * The input voicing is something like 1 3 5 7 9 11 13. You can add a series
104      * of + or - characters to jump octaves.
105      * 
106      * <pre>
107      * MIDITrack nl = ChordVoicing.parse(&quot;1 5 ++3&quot;, c);
108      * </pre>
109      * 
110      * If you give an invalid voicing it just complains. e.g. "1 3 5 7" for a
111      * triad. The seventh will be ignored.
112      * 
113      * 
114      * @param s
115      *            a Voicing String
116      * @param c
117      *            a Chord instance
118      * @return a MIDITrack
119      * @see MIDITrack
120      */
121     private static MIDITrack parse(String s, Chord c) {
122         int octaveOffset = 0;
123         boolean repeatedRoot = false;
124         boolean repeatedThird = false;
125         boolean repeatedFifth = false;
126         boolean repeatedSeventh = false;
127         boolean repeatedNinth = false;
128         boolean repeatedEleventh = false;
129         boolean repeatedThirteenth = false;
130 
131         MIDITrack track = new MIDITrack();
132         Scanner scanner = new Scanner(s.trim());
133         while (scanner.hasNext()) {
134             octaveOffset = 0;
135             String d = scanner.next().trim();
136             while (d.indexOf('+') != -1) {
137                 octaveOffset++;
138                 d = d.substring(1);
139             }
140             while (d.indexOf('-') != -1) {
141                 octaveOffset--;
142                 d = d.substring(1);
143             }
144 
145             int i = Integer.parseInt(d);
146             int n = 0;
147             switch (i) {
148             case 1:
149                 n = c.getRoot();
150                 // if the voicing is something like: 1 3 5 1 3 5 the root is
151                 // repeated and the oct
152                 // needs to be bumped up.
153                 if (repeatedRoot) {
154                     octaveOffset++;
155                     repeatedRoot = false;
156                 }
157                 addMIDINote(octaveOffset, track, n, c);
158                 repeatedRoot = true;
159                 break;
160             case 3:
161                 n = c.getThird();
162                 // what about 1 3 3 5?
163                 // nah, would be best to make the exception for 1
164                 // and make the others specify the jump.
165                 // so, 1 3 + 3 5
166 
167                 // if (repeatedThird) {
168                 // octaveOffset++;
169                 // }
170                 addMIDINote(octaveOffset, track, n, c);
171                 repeatedThird = true;
172                 break;
173             case 5:
174                 n = c.getFifth();
175                 addMIDINote(octaveOffset, track, n, c);
176                 repeatedFifth = true;
177                 break;
178             case 7:
179                 if (c.hasSeventh()) {
180                     n = c.getSeventh();
181                     addMIDINote(octaveOffset, track, n, c);
182                     repeatedSeventh = true;
183                 } else {
184                     System.err.printf("%s does not have a %d%n", c
185                             .getDisplayName(), i);
186                     logger.error(String.format("%s does not have a %d%n", c
187                             .getDisplayName(), i));
188                 }
189                 break;
190             case 9:
191                 if (c.hasNinth()) {
192                     n = c.getNinth();
193                     addMIDINote(octaveOffset, track, n, c);
194                     repeatedNinth = true;
195                 } else {
196                     System.err.printf("%s does not have a %d%n", c
197                             .getDisplayName(), i);
198                     logger.error(String.format("%s does not have a %d%n", c
199                             .getDisplayName(), i));
200                 }
201                 break;
202             case 11:
203                 if (c.hasEleventh()) {
204                     n = c.getEleventh();
205                     addMIDINote(octaveOffset, track, n, c);
206                     repeatedEleventh = true;
207                 } else {
208                     System.err.printf("%s does not have a %d%n", c
209                             .getDisplayName(), i);
210                     logger.error(String.format("%s does not have a %d%n", c
211                             .getDisplayName(), i));
212                 }
213                 break;
214             case 13:
215                 if (c.hasThirteenth()) {
216                     n = c.getThirteenth();
217                     addMIDINote(octaveOffset, track, n, c);
218                     repeatedThirteenth = true;
219                 } else {
220                     System.err.printf("%s does not have a %d%n", c
221                             .getDisplayName(), i);
222                     logger.error(String.format("%s does not have a %d%n", c
223                             .getDisplayName(), i));
224                 }
225                 break;
226                 default :
227                     logger.debug("Don't know {}", i);
228                     break;
229             }
230         }
231         scanner.close();
232 
233         return track;
234     }

236     private static void addMIDINote(int octaveOffset, MIDITrack notelist,
237             int n, Chord chord) {
238         logger.debug(String.format("adding note %d with offset %d%n", n,
239                 octaveOffset));
240         if (octaveOffset > 0) {
241             while (octaveOffset-- > 0) {
242                 n += 12;
243             }
244         } else if (octaveOffset < 0) {
245             while (octaveOffset++ < 0) {
246                 n -= 12;
247             }
248         }
249         MIDINote mn = new MIDINote(n, chord.getStartBeat(), chord.getDuration());
250         if (logger.isDebugEnabled()) {
251             logger.debug(String.format(
252                     "pitch %d created and added midi note %s", n, mn));
253         }
254         notelist.add(mn);
255     }

262     // drop 2 is drop the second highest pitch an octave
263     public static String getDropString(Chord chord, int drop) {
264         StringBuilder sb = new StringBuilder();
265         MIDITrack notelist = chord.createMIDITrack();
266         notelist.sortByAscendingPitches();
267         int size = notelist.size();
268         int index = 0;
269         if (drop > 0)
270             index = size - drop;
271         else
272             index = 0;
273         MIDINote n = notelist.get(index);
274         logger.debug("Dropping " + n);
275         if (n.getMidiNumber() >= 12) {
276             n.setPitch(n.getPitch().transpose(-12));
277         } else {
278             logger.error("Too low to drop " + n);
279         }
280 
281         for (MIDINote note : notelist) {
282             if (note == n && index != 0) {
283                 sb.append("-");
284             }
285             int mn = note.getMidiNumber();
286             int degree = chord.pitchToChordDegree(mn);
287             logger.debug(String.format("chord %s mn %d degree %d", chord,
288                     mn, degree));
289             sb.append(degree).append(' ');
290         }
291 
292         return sb.toString().trim();
293     }

296      * pretty much closed position.
297      * 
298      * @param chord the chord
299      * @param inversion the inversion
300      * @return a String with the inversion
301      */
302     public String getInversion(Chord chord, int inversion) {
303         StringBuilder sb = new StringBuilder();
304         String[] sa = chord.getDefaultVoicingString().split("\\s");
305         // String[] sa = this.voicing.split("\\s");
306         String[] dest = new String[sa.length];
307 
308        // int croot = chord.getRoot();
309         // int coct = croot / 12;
310         // logger.debug(String.format("croot %d coct is %d", croot, coct));
311 
312         // rotate
313         System.arraycopy(sa, inversion, dest, 0, sa.length - inversion);
314         System.arraycopy(sa, 0, dest, sa.length - inversion, inversion);
315 
316         String firstPitchString = dest[0];
317         if (firstPitchString.indexOf('+') != -1) {
318             firstPitchString = firstPitchString.substring(firstPitchString
319                     .lastIndexOf('+') + 1);
320         }
321         if (firstPitchString.indexOf('-') != -1) {
322             firstPitchString = firstPitchString.substring(firstPitchString
323                     .lastIndexOf('-') + 1);
324         }
325         int firstPitch = chord.getChordDegree(Integer
326                 .parseInt(firstPitchString));
327         logger.debug(String.format("first pitch is %d", firstPitch));
328         // firstPitch = croot;
329         logger.debug(String.format("first pitch is %d", firstPitch));
330         // int firstPitchOct = firstPitch / 12;
331         logger.debug(String.format("%d is the first pitch", firstPitch));
332         for (String s : dest) {
333             if (s.indexOf('+') != -1) {
334                 s = s.substring(s.lastIndexOf('+') + 1);
335             }
336             if (s.indexOf('-') != -1) {
337                 s = s.substring(s.lastIndexOf('-') + 1);
338             }
339             int i = Integer.parseInt(s);
340             int p = chord.getChordDegree(i);
341             while (p < firstPitch && p < Pitch.MAX) {
342                 logger.debug(String.format("%d bumped up an octave to %d, %d",
343                         p, p + 12, firstPitch));
344                 p += 12;
345                 sb.append('+');
346 
347             }
348             logger.debug(String.format("%d is the pitch degree for %s", p, s));
349             //
350             // int nPlus = (p /12) - coct;
351             // logger.debug(String.format("root oct %d nplus is %d", (p /12),
352             // nPlus));
353 
354             sb.append(s).append(' ');
355         }
356 
357         if (logger.isDebugEnabled()) {
358             String s = String.format("inversion %d is this voicing %s",
359                     inversion, sb.toString());
360             logger.debug(s);
361         }
362         return sb.toString();
363     }
