156             final String key, final Scale scale, final int beatsPerMeasure) {
157         final StringBuilder sb = new StringBuilder();

160             // String ps = PitchFormat.getPitchString(chord.getRoot());
161             // String roman = RomanChordParser.pitchNameToRoman(scale, ps, key);
162             final String roman = scale.getRoman(key, chord);

164             // String sym = chord.getSymbol();
165             if (beat > beatsPerMeasure) {
166                 beat = 1d;
167                 sb.append('|');
168             }
169             sb.append(roman).append(' ');
170 
171             // if (sym.equals("maj") || sym.equals("")) {
172             // sb.append(roman).append(' ');
173             // } else {
174             // sb.append(roman).append(' ');
175             // //sb.append(roman).append(chord.getSymbol()).append(' ');
176             // }
177 
178             final double chordDur = chord.getDuration();
179             logger.debug(String.format(
180                     "beat %f chord dur %f", beat, chordDur));
181 
182             if (chordDur > 1d) {
183                 for (int i = 0; i < chordDur - 1; i++) {
184                     sb.append("/ ");
185                 }
186             }
187 
188             beat += chordDur;
189         }
190         return sb.toString().trim();

232     /**
233      * Removes beats between chords.
234      * 
235      * @param gap
236      *            duration between chords
237      * @return the total
238      */
239     public double closeTimeline(final double gap) {
240         final int size = this.chords.size();
241         final Chord n = this.chords.get(0);
242         double s = n.getStartBeat();
243         final double begin = s;
244         double d = n.getDuration();
245         double total = d - s;
246         for (int i = 1; i < size; i++) {
247             final Chord nn = this.chords.get(i);
248             nn.setStartBeat(s + d + gap);
249             s = nn.getStartBeat();
250             d = nn.getDuration();
251             total = (s + d) - begin;
252         }
253         return total;
254     }

264     /**
265      * Return the Chord that matches if the beat is between the Chord's start
266      * and end.
267      * 
268      * @param beat
269      *            a beat
270      * @return a Chord or null if not chords at beat
271      */
272     public Chord getChordAtBeat(final double beat) {
273         logger.debug("looking for  sb {}", beat);
274         for (final Chord n : this) {
275             final double s = n.getStartBeat();
276             final double e = n.getEndBeat();
277             logger.debug("checking sb {} of chord {}", s, n);
278             Range<Double> range = Range.between(s, e - .0001);
279             logger.debug("range {}", range);
280             if (range.contains(beat)) {
281                 logger.debug("returning chord {}", n);
282                 return n;
283             }
284         }
285         return null;
286     }

373     public void sortByStartBeat() {
374         if (this.chords == null) {
375             return;
376         }
377         final Comparator<Chord> c = new Comparator<Chord>() {
378             @Override
379             public int compare(final Chord o1, final Chord o2) {
380                 if(Math.abs(o1.getStartBeat() - o2.getStartBeat()) < .0000001) {
381                     return 0;
382                 }
383                 if (o1.getStartBeat() < o2.getStartBeat()) {
384                     return -1;
385                 }
386                 if (o1.getStartBeat() > o2.getStartBeat()) {
387                     return 1;
388                 }
389                 return 0;
390             }
391         };
392         Collections.sort(this.chords, c);
393     }

452         sb.append("<tbody>\n");

454             if (count == 0) {
455                 sb.append("<tr>");
456             }
457 
458             sb.append("<td>");
459             // the regexp eats the |
460             sb.append(s.trim());
461             // sb.append(s.trim()).append('|');
462             sb.append("</td>");

464             if (!(++count < measuresPerRow)) {
465                 sb.append("</tr>\n");
466                 count = 0;
467             }
468         }
469         sb.append("</tr>");
470         sb.append("\n</tbody></table>\n");
471         sb.append("</body></html>");

557     public String toStd(final boolean omitSymbol, final boolean useSlashes) {
558         final StringBuilder sb = new StringBuilder();
559         double beat = 1d;
560 
561         for (final Chord chord : this) {
562             if (beat > this.beatsPerMeasure) {
563                 beat = 1d;
564                 sb.append("| ");
565             }
566             if (omitSymbol) {
567                 sb.append(PitchFormat.getPitchString(chord.getRoot())).append(
568                         ' ');
569             } else {
570                 sb.append(chord.getDisplayName()).append(' ');
571             }
572             final double chordDur = chord.getDuration();
573             if (ChordProgression.logger.isDebugEnabled()) {
574                 ChordProgression.logger.debug(String.format(
575                         "beat %f chord dur %f", beat, chordDur));
576             }
577             if (chordDur > 1d) {
578                 if (chordDur > this.beatsPerMeasure) {
579                     for (int len = 0; len < chordDur; len += this.beatsPerMeasure) {
580                         if (useSlashes) {
581                             for (int i = 0; i < this.beatsPerMeasure - 1; i++) {
582                                 sb.append("/ ");
583                             }
584                         }
585                         if (logger.isDebugEnabled()) {
586                             logger.debug(String.format("len %d chordDur %f",
587                                     len, chordDur));
588                         }
589                         if (len < chordDur - beatsPerMeasure) {
590                             sb.append("| ");
591                             sb.append(chord.getDisplayName()).append(' ');
592                         }
593                         beat = 1d;
594                     }
595                 } else {
596                     if (useSlashes) {
597                         for (int i = 0; i < chordDur - 1; i++) {
598                             sb.append("/ ");
599                         }
600                     }
601                 }
602             }
603             beat += chordDur;
604         }
605         sb.append("|");
606         return sb.toString().trim();
607     }

629     public ChordProgression append(ChordProgression chordProgression) {
630         this.closeTimeline();
631         chordProgression.closeTimeline();
632         double startBeat = this.getEndBeat();
633 
634         // avoid a concurrent modification exception
635         if (this == chordProgression) {
636             List<Chord> list = new ArrayList<Chord>();
637             list.addAll(this.chords);
638             for (Chord c : chordProgression.chords) {
639                 Chord dupe = new Chord(c);
640                 dupe.setStartBeat(startBeat);
641                 list.add(dupe);
642                 startBeat += dupe.getDuration();
643             }
644             this.setChords(list);
645             return this;
646         }
647 
648         for (Chord c : chordProgression.chords) {
649             Chord dupe = new Chord(c);
650             dupe.setStartBeat(startBeat);
651             this.add(dupe);
652             startBeat += dupe.getDuration();
653         }
654         return this;
655     }
