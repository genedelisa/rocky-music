137     public static void writeChordProgressionXML(
138             ChordProgression chordProgression, OutputStream out) {
139         logger.debug("enter writeChordXML");
140         DocumentBuilderFactory dbf = null;
141         DocumentBuilder db = null;
142         try {
143             dbf = DocumentBuilderFactory.newInstance();
144             dbf.setValidating(false);
145             // dbf.setSchema()
146             db = dbf.newDocumentBuilder();
147         } catch (ParserConfigurationException e) {
148             e.printStackTrace();
149             return;
150         }
151         Document doc = (Document) db.newDocument();
152         String namespaceURI = "http://www.rockhoppertech.com/music/chordprogression";
153         Element root = doc.createElementNS(namespaceURI, "chordprogression");
154         doc.appendChild(root);
155 
156         // DecimalFormat fmt = new DecimalFormat("#,###,###,###.##");
157         Element chordElement = null;
158         for (Chord chord : chordProgression) {
159             chordElement = doc.createElement("chord");
160             root.appendChild(chordElement);
161             chordElement.setAttribute("displayName", chord.getDisplayName());
162 
163             Element octave = doc.createElement("octave");
164             octave.appendChild(doc.createTextNode("" + chord.getOctave()));
165             chordElement.appendChild(octave);
166 
167             Element startBeat = doc.createElement("startBeat");
168             startBeat
169                     .appendChild(doc.createTextNode("" + chord.getStartBeat()));
170             chordElement.appendChild(startBeat);
171 
172             Element duration = doc.createElement("duration");
173             duration.appendChild(doc.createTextNode("" + chord.getDuration()));
174             chordElement.appendChild(duration);
175 
176             Element voicing = doc.createElement("voicing");
177             chordElement.appendChild(voicing);
178             String[] voicingDegrees = chord.getChordVoicing()
179                     .split("\\s");
180             for (String degree : voicingDegrees) {
181                 Element degreeEl = doc.createElement("degree");
182                 degreeEl.appendChild(doc.createTextNode(degree));
183                 voicing.appendChild(degreeEl);
184             }
185         }
186 
187         doc.getDocumentElement().normalize();
188 
189         try {
190             DOMSource domSource = new DOMSource(doc);
191             StreamResult streamResult = new StreamResult(out);
192             TransformerFactory tf = TransformerFactory.newInstance();
193             Transformer transformer = tf.newTransformer();
194             Properties oprops = new Properties();
195             oprops.put(OutputKeys.INDENT, "yes");
196             oprops.put(OutputKeys.METHOD, "xml");
197             oprops.put(OutputKeys.OMIT_XML_DECLARATION, "no");
198             transformer.setOutputProperties(oprops);
199             transformer.transform(domSource, streamResult);
200         } catch (TransformerException e) {
201             System.err.println(e);
202         }
203     }

206             throws UnknownChordException {
207         DocumentBuilder db = null;
208         try {
209             DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
210             dbf.setCoalescing(true);

214             dbf.setNamespaceAware(true);

223         }
224 
225         Document doc = null;
226         try {
227             InputSource is = new InputSource(new StringReader(xml));
228             doc = db.parse(is);
229             doc.normalize();
230         } catch (SAXParseException spe) {
231             logger.error(spe.getLocalizedMessage(), spe);
232             Exception x = spe.getException();
233             if (x != null) {
234                 logger.error(x.getLocalizedMessage(), x);
235             }
236             return null;

256         logger.debug("parsing chords document element " + name);
257         Chord chord = null;
258         List<Chord> list = new ArrayList<Chord>();
259         if (name.equalsIgnoreCase("ChordProgression")) {
260             NodeList chords = e.getElementsByTagName("chord");
261             if (chords != null) {
262                 int length = chords.getLength();
263                 for (int i = 0; i < length; i++) {
264                     Node nl = chords.item(i);
265                     logger.debug("parsing chord ");
266                     chord = parseChordProgressionChord(nl);
267                     list.add(chord);
268                 }
269             } else {
270                 System.err.println("no chords!");
271             }
272         }
273 
274         return new ChordProgression(list, "");
275     }

277     public static void writeChordXML(OutputStream out) {
278         logger.debug("enter writeChordXML");
279         DocumentBuilderFactory dbf = null;
280         DocumentBuilder db = null;
281         try {
282             dbf = DocumentBuilderFactory.newInstance();
283             dbf.setValidating(false);
284             db = dbf.newDocumentBuilder();
285         } catch (ParserConfigurationException e) {
286             e.printStackTrace();
287             return;
288         }
289         Document doc = (Document) db.newDocument();
290 
291         Element chords = doc.createElement("chords");
292         doc.appendChild(chords);
293 
294         // Element extends Node
295 
296         Set<String> list = ChordFactory.getSymbolNames();
297 
298         for (String symbol : list) {
299             Chord mc = null;
300             try {
301                 mc = ChordFactory.getChordBySymbol(symbol);
302             } catch (UnknownChordException e) {
303                 e.printStackTrace();
304                 logger.error(e.getLocalizedMessage(), e);
305                 return;
306             }
307             Element chord = doc.createElement("chord");
308             chords.appendChild(chord);
309             chord.setAttribute("name", mc.getSymbol());
310 
311             Element intervals = doc.createElement("intervals");
312             chord.appendChild(intervals);
313             for (Integer i : mc.getIntervals()) {
314                 Element intervalel = doc.createElement("interval");
315                 String interval = i.toString();
316                 intervalel.appendChild(doc.createTextNode(interval));
317                 intervals.appendChild(intervalel);
318             }
319 
320             Element spelling = doc.createElement("spelling");
321             chord.appendChild(spelling);
322             String s = mc.getSpelling();
323             String[] spellings = s.split("\\s");
324             for (String degree : spellings) {
325                 Element degreeEl = doc.createElement("degree");
326                 degreeEl.appendChild(doc.createTextNode(degree));
327                 spelling.appendChild(degreeEl);
328             }
329 
330             String descr = mc.getDescription();
331             if (descr != null) {
332                 Element el = doc.createElement("description");
333                 chord.appendChild(el);
334                 el.appendChild(doc.createTextNode(descr));
335             }
336         }
337         logger.debug("writing to stdout");
338         try {
339             DOMSource domSource = new DOMSource(doc);
340             StreamResult streamResult = new StreamResult(out);
341             TransformerFactory tf = TransformerFactory.newInstance();
342             Transformer transformer = tf.newTransformer();
343             Properties oprops = new Properties();
344             oprops.put(OutputKeys.INDENT, "yes");
345             oprops.put(OutputKeys.METHOD, "xml");
346             oprops.put(OutputKeys.OMIT_XML_DECLARATION, "no");
347             oprops.put(OutputKeys.DOCTYPE_PUBLIC,
348                     "-//Rockhopper Technologies//DTD chords 1.0//EN");
349             oprops.put(OutputKeys.DOCTYPE_SYSTEM,
350                     "http://www.rockhoppertech.com/dtds/chords.dtd");
351 
352             transformer.setOutputProperties(oprops);
353             transformer.transform(domSource, streamResult);
354         } catch (TransformerException e) {
355             System.err.println(e);
356         }
357         logger.debug("leave writexml chord");
358     }

624     private static Chord parseChordProgressionChord(Node chord)
625             throws UnknownChordException {
626         if (chord == null) {
627             System.err.println("parseChord: node is null");
628             return null;
629         }
630         if (chord.getNodeName().equals("chord") == false) {
631             System.err.println("parseChord: node is not a chord");
632             return null;
633         }
634         NamedNodeMap pa = chord.getAttributes();
635         Node ni = pa.getNamedItem("displayName");
636         if (ni == null) {
637             throw new IllegalArgumentException("Invalid chord");
638         }
639         String displayName = ni.getNodeValue();
640         String voicing = null;
641         int octave = 0;
642         double startBeat = 1d;
643         double duration = 4d;
644 
645         /*
646          * <chordprogression
647          * xmlns="http://www.rockhoppertech.com/music/chordprogression"> <chord
648          * displayName="C"> <octave>5</octave> <startBeat>1.0</startBeat>
649          * <duration>2.0</duration> <voicing> <degree>1</degree>
650          * <degree>3</degree> <degree>5</degree> </voicing> </chord>
651          * 
652          * </chordprogression>
653          */
654         if (chord.hasChildNodes()) {
655             NodeList sc = chord.getChildNodes();
656             int len = sc.getLength();
657             for (int i = 0; i < len; i++) {
658                 Node nn = sc.item(i);
659                 if (nn.getNodeName().equalsIgnoreCase("voicing")) {
660                     voicing = parseVoicing(nn);
661                 }
662                 if (nn.getNodeName().equalsIgnoreCase("octave")) {
663                     NodeList cn = nn.getChildNodes();
664                     Node on = cn.item(0);
665                     octave = Integer.parseInt(on.getTextContent());
666                 }
667                 if (nn.getNodeName().equalsIgnoreCase("startBeat")) {
668                     NodeList cn = nn.getChildNodes();
669                     Node on = cn.item(0);
670                     startBeat = Double.parseDouble(on.getTextContent());
671                 }
672                 if (nn.getNodeName().equalsIgnoreCase("duration")) {
673                     NodeList cn = nn.getChildNodes();
674                     Node on = cn.item(0);
675                     duration = Double.parseDouble(on.getTextContent());
676                 }
677             }
678         }
679         Chord mc = ChordFactory.getChordByFullSymbol(displayName);
680         if (voicing != null) {
681             mc.setChordVoicing(voicing);
682         }
683         mc.setOctave(octave);
684         mc.setStartBeat(startBeat);
685         mc.setDuration(duration);
686 
687         return mc;
688     }

691      * Attempt to parse each chord element and add it to the ChordFactory.
692      * Ignores the spelling if the intervals are specified.
693      * 
694      * @param chord
695      */
696     private static Chord parseChord(Node chord) {
697         if (chord == null) {
698             System.err.println("parseChord: node is null");
699             return null;
700         }
701         if (chord.getNodeName().equals("chord") == false) {
702             System.err.println("parseChord: node is not a chord");
703             return null;
704         }
705         NamedNodeMap pa = chord.getAttributes();
706         Node ni = pa.getNamedItem("name");
707         if (ni == null) {
708             throw new IllegalArgumentException("Invalid chord");
709         }
710         String symbol = ni.getNodeValue();
711         Integer[] intervals = null;
712         String spelling = null;
713         String description = null;
714         String[] aliases = null;
715         String voicing = null;
716 
717         if (chord.hasChildNodes()) {
718             NodeList sc = chord.getChildNodes();
719             int len = sc.getLength();
720             for (int i = 0; i < len; i++) {
721                 Node nn = sc.item(i);
722                 if (nn.getNodeName().equalsIgnoreCase("intervals")) {
723                     intervals = parseIntervals(nn);
724                 }
725                 if (nn.getNodeName().equalsIgnoreCase("spelling")) {
726                     spelling = parseSpelling(nn);
727                 }
728                 if (nn.getNodeName().equalsIgnoreCase("description")) {
729                     description = nn.getTextContent();
730                     description = description.trim();
731                 }
732                 if (nn.getNodeName().equalsIgnoreCase("aliases")) {
733                     aliases = parseAliases(nn);
734                 }
735                 if (nn.getNodeName().equalsIgnoreCase("voicing")) {
736                     voicing = parseVoicing(nn);
737                 }
738             }
739         }
740         Chord mc = null;
741 
742         if (intervals != null) {
743             mc = new Chord(symbol, intervals, description);
744         } else {
745             mc = new Chord(symbol, spelling, description);
746         }
747         if (voicing != null) {
748             mc.setChordVoicing(voicing);
749         }
750 
751         if (aliases == null) {
752             ChordFactory.registerChord(mc, mc.getSymbol());
753             if (logger.isDebugEnabled()) {
754                 String s = String.format("registering chord no alias '%s'", mc);
755                 logger.debug(s);
756             }
757         } else {
758             ChordFactory.registerChord(mc, mc.getSymbol());
759             for (String alias : aliases) {
760                 mc.addAlias(alias);
761                 ChordFactory.registerChord(mc, alias);
762                 if (logger.isDebugEnabled()) {
763                     String s = String.format("registering chord alias '%s'",
764                             alias);
765                     logger.debug(s);
766                 }
767             }
768         }
769         return mc;
770         /*
771          * System.out.println(mc); System.out.println(intervals);
772          * System.out.println(spelling);
773          */
774     }

867         String namespaceURI = "http://www.rockhoppertech.com/music/chords";
868         Element root = doc.createElementNS(namespaceURI, "chords");
869         doc.appendChild(root);
870 
871         // DecimalFormat fmt = new DecimalFormat("#,###,###,###.##");
872         Element chordElement = null;

874         root.appendChild(chordElement);

876

878         chordElement.appendChild(intervalsElement);

882             intervalsElement.appendChild(intervalElement);

905                 .split("\\s");

908             degreeEl.appendChild(doc.createTextNode(degree));
909             voicing.appendChild(degreeEl);
